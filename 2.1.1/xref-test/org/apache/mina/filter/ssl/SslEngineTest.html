51      private static SSLContext createSSLContext() throws IOException, GeneralSecurityException {
52          char[] passphrase = "password".toCharArray();
53  
54          SSLContext ctx = SSLContext.getInstance("TLS");
55          KeyManagerFactory kmf = KeyManagerFactory.getInstance(KEY_MANAGER_FACTORY_ALGORITHM);
56          TrustManagerFactory tmf = TrustManagerFactory.getInstance(KEY_MANAGER_FACTORY_ALGORITHM);
57  
58          KeyStore ks = KeyStore.getInstance("JKS");
59          KeyStore ts = KeyStore.getInstance("JKS");
60  
61          ks.load(SslTest.class.getResourceAsStream("keystore.sslTest"), passphrase);
62          ts.load(SslTest.class.getResourceAsStream("truststore.sslTest"), passphrase);
63  
64          kmf.init(ks, passphrase);
65          tmf.init(ts);
66          ctx.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
67  
68          return ctx;
69      }

72      /**
73       * Decrypt the incoming buffer and move the decrypted data to an
74       * application buffer.
75       */
76      private SSLEngineResult unwrap(SSLEngine sslEngine, IoBuffer inBuffer, IoBuffer outBuffer) throws SSLException {
77          // We first have to create the application buffer if it does not exist
78          if (outBuffer == null) {
79              outBuffer = IoBuffer.allocate(inBuffer.remaining());
80          } else {
81              // We already have one, just add the new data into it
82              outBuffer.expand(inBuffer.remaining());
83          }
84  
85          SSLEngineResult res;
86          Status status;
87          HandshakeStatus localHandshakeStatus;
88  
89          do {
90              // Decode the incoming data
91              res = sslEngine.unwrap(inBuffer.buf(), outBuffer.buf());
92              status = res.getStatus();
93  
94              // We can be processing the Handshake
95              localHandshakeStatus = res.getHandshakeStatus();
96  
97              if (status == SSLEngineResult.Status.BUFFER_OVERFLOW) {
98                  // We have to grow the target buffer, it's too small.
99                  // Then we can call the unwrap method again
100                 int newCapacity = sslEngine.getSession().getApplicationBufferSize();
101                 
102                 if (inBuffer.remaining() >= newCapacity) {
103                     // The buffer is already larger than the max buffer size suggested by the SSL engine.
104                     // Raising it any more will not make sense and it will end up in an endless loop. Throwing an error is safer
105                     throw new SSLException("SSL buffer overflow");
106                 }
107 
108                 inBuffer.expand(newCapacity);
109                 continue;
110             }
111         } while (((status == SSLEngineResult.Status.OK) || (status == SSLEngineResult.Status.BUFFER_OVERFLOW))
112                 && ((localHandshakeStatus == SSLEngineResult.HandshakeStatus.NOT_HANDSHAKING) || 
113                         (localHandshakeStatus == SSLEngineResult.HandshakeStatus.NEED_UNWRAP)));
114 
115         return res;
116     }

119     private SSLEngineResult.Status unwrapHandshake(SSLEngine sslEngine, IoBuffer appBuffer, IoBuffer netBuffer) throws SSLException {
120         // Prepare the net data for reading.
121         if ((appBuffer == null) || !appBuffer.hasRemaining()) {
122             // Need more data.
123             return SSLEngineResult.Status.BUFFER_UNDERFLOW;
124         }
125 
126         SSLEngineResult res = unwrap(sslEngine, appBuffer, netBuffer);
127         HandshakeStatus handshakeStatus = res.getHandshakeStatus();
128 
129         //checkStatus(res);
130 
131         // If handshake finished, no data was produced, and the status is still
132         // ok, try to unwrap more
133         if ((handshakeStatus == SSLEngineResult.HandshakeStatus.FINISHED)
134                 && (res.getStatus() == SSLEngineResult.Status.OK)
135                 && appBuffer.hasRemaining()) {
136             res = unwrap(sslEngine, appBuffer, netBuffer);
137 
138             // prepare to be written again
139             if (appBuffer.hasRemaining()) {
140                 appBuffer.compact();
141             } else {
142                 appBuffer.free();
143                 appBuffer = null;
144             }
145         } else {
146             // prepare to be written again
147             if (appBuffer.hasRemaining()) {
148                 appBuffer.compact();
149             } else {
150                 appBuffer.free();
151                 appBuffer = null;
152             }
153         }
154 
155         return res.getStatus();
156     }

169     /**
170      * Perform any handshaking processing.
171      */
172     /* no qualifier */HandshakeStatus handshake(SSLEngine sslEngine, IoBuffer appBuffer, IoBuffer netBuffer ) throws SSLException {
173         SSLEngineResult result;
174         HandshakeStatus handshakeStatus = sslEngine.getHandshakeStatus();
175 
176         for (;;) {
177             switch (handshakeStatus) {
178             case FINISHED:
179                 //handshakeComplete = true;
180                 return handshakeStatus;
181 
182             case NEED_TASK:
183                 //handshakeStatus = doTasks();
184                 break;
185 
186             case NEED_UNWRAP:
187                 // we need more data read
188                 SSLEngineResult.Status status = unwrapHandshake(sslEngine, appBuffer, netBuffer);
189                 handshakeStatus = sslEngine.getHandshakeStatus();
190 
191                 return handshakeStatus;
192 
193             case NEED_WRAP:
194                 result = sslEngine.wrap(emptyBuffer.buf(), netBuffer.buf());
195 
196                 while ( result.getStatus() == SSLEngineResult.Status.BUFFER_OVERFLOW ) {
197                     netBuffer.capacity(netBuffer.capacity() << 1);
198                     netBuffer.limit(netBuffer.capacity());
199 
200                     result = sslEngine.wrap(emptyBuffer.buf(), netBuffer.buf());
201                 }
202 
203                 netBuffer.flip();
204                 return result.getHandshakeStatus();
205 
206             case NOT_HANDSHAKING:
207                 result = sslEngine.wrap(emptyBuffer.buf(), netBuffer.buf());
208 
209                 while ( result.getStatus() == SSLEngineResult.Status.BUFFER_OVERFLOW ) {
210                     netBuffer.capacity(netBuffer.capacity() << 1);
211                     netBuffer.limit(netBuffer.capacity());
212 
213                     result = sslEngine.wrap(emptyBuffer.buf(), netBuffer.buf());
214                 }
215 
216                 netBuffer.flip();
217                 handshakeStatus = result.getHandshakeStatus();
218                 return handshakeStatus;
219 
220             default:
221                 throw new IllegalStateException("error");
222             }
223         }
224     }

227     /**
228      * Do all the outstanding handshake tasks in the current Thread.
229      */
230     private SSLEngineResult.HandshakeStatus doTasks(SSLEngine sslEngine) {
231         /*
232          * We could run this in a separate thread, but I don't see the need for
233          * this when used from SSLFilter. Use thread filters in MINA instead?
234          */
235         Runnable runnable;
236         while ((runnable = sslEngine.getDelegatedTask()) != null) {
237             //Thread thread = new Thread(runnable);
238             //thread.start();
239             runnable.run();
240         }
241         return sslEngine.getHandshakeStatus();
242     }

245     private HandshakeStatus handshake(SSLEngine sslEngine, HandshakeStatus expected, 
246         IoBuffer inBuffer, IoBuffer outBuffer) throws SSLException {
247         HandshakeStatus handshakeStatus = handshake(sslEngine, inBuffer, outBuffer);
248 
249         if ( handshakeStatus != expected) {
250             fail();
251         }
252         
253         return handshakeStatus;
254     }

257     @Test
258     public void testSSL() throws Exception {
259         // Initialise the client SSLEngine
260         SSLContext sslContextClient = createSSLContext();
261         SSLEngine sslEngineClient = sslContextClient.createSSLEngine();
262         int packetBufferSize = sslEngineClient.getSession().getPacketBufferSize();
263         inNetBufferClient = IoBuffer.allocate(packetBufferSize).setAutoExpand(true);
264         outNetBufferClient = IoBuffer.allocate(packetBufferSize).setAutoExpand(true);
265         
266         sslEngineClient.setUseClientMode(true);
267 
268         // Initialise the Server SSLEngine
269         SSLContext sslContextServer = createSSLContext();
270         SSLEngine sslEngineServer = sslContextServer.createSSLEngine();
271         packetBufferSize = sslEngineServer.getSession().getPacketBufferSize();
272         inNetBufferServer = IoBuffer.allocate(packetBufferSize).setAutoExpand(true);
273         outNetBufferServer = IoBuffer.allocate(packetBufferSize).setAutoExpand(true);
274         
275         sslEngineServer.setUseClientMode(false);
276 
277         HandshakeStatus handshakeStatusClient = sslEngineClient.getHandshakeStatus();
278         HandshakeStatus handshakeStatusServer = sslEngineServer.getHandshakeStatus();
279         
280         // Start the server
281         handshakeStatusServer = handshake(sslEngineServer, HandshakeStatus.NEED_UNWRAP, inNetBufferServer, outNetBufferServer);
282         
283         // Now start the client
284         handshakeStatusClient = handshake(sslEngineClient, HandshakeStatus.NEED_UNWRAP, inNetBufferClient, outNetBufferClient);
285         
286         // 'Read' the CLIENT_HELLO to the server
287         handshakeStatusServer = handshake(sslEngineServer, HandshakeStatus.NEED_TASK, outNetBufferClient, outNetBufferServer);
288 
289         // Create the SERVER_HELLO message
290         handshakeStatusServer = doTasks(sslEngineServer);
291         
292         // We should get back the message  
293         if ( handshakeStatusServer != HandshakeStatus.NEED_WRAP) {
294             fail();
295         }
296         
297         // 'Send' the SERVER_HELLO message to the client
298         outNetBufferServer.clear();
299         handshakeStatusServer = handshake(sslEngineServer, HandshakeStatus.NEED_UNWRAP, null, outNetBufferServer);
300         
301         // 'Read' the SERVER_HELLO message on the client
302         handshakeStatusClient = handshake(sslEngineClient, HandshakeStatus.NEED_TASK, outNetBufferServer, inNetBufferClient);
303         
304         // Create the  message
305         handshakeStatusClient = doTasks(sslEngineClient);
306         
307         // We should get back the message  
308         if ( handshakeStatusClient != HandshakeStatus.NEED_WRAP) {
309             fail();
310         }
311         
312         // 'Send' the SERVER_HELLO message to the client
313         outNetBufferClient.clear();
314         handshakeStatusClient = handshake(sslEngineClient, HandshakeStatus.NEED_WRAP, null, outNetBufferClient);
315         
316         // 'Send' the CLIENT_KEY_EXCHANGE message to the server
317         outNetBufferClient.clear();
318         handshakeStatusClient = handshake(sslEngineClient, HandshakeStatus.NEED_WRAP, null, outNetBufferClient);
319         
320         // 'Send' the ALERT message to the server
321         outNetBufferClient.clear();
322         handshakeStatusClient = handshake(sslEngineClient, HandshakeStatus.NEED_UNWRAP, null, outNetBufferClient);
323     }
