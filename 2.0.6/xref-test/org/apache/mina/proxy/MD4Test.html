34  /**
35   * MD4Test.java - JUnit testcase that tests the rfc 1320 test suite.
36   * @see <a href="http://www.ietf.org/rfc/rfc1320.txt">RFC 1320</a>
37   * 
38   * @author <a href="http://mina.apache.org">Apache MINA Project</a>
39   * @since MINA 2.0.0-M3
40   */
41  public class MD4Test {
42  
43      /**
44       * {@inheritDoc}
45       */
46      @Before
47      public void setUp() throws Exception {
48          if (Security.getProvider(MD4Provider.PROVIDER_NAME) == null) {
49              System.out.print("Adding MINA provider...");
50              Security.addProvider(new MD4Provider());
51              //System.out.println(" [Ok]");
52          }
53      }
54  
55      /**
56       * Test suite for the MD4 algorithm. 
57       */
58      @Test
59      public void testRFCVectors() throws NoSuchAlgorithmException, NoSuchProviderException {
60          MessageDigest md4 = MessageDigest.getInstance("MD4", MD4Provider.PROVIDER_NAME);
61          doTest(md4, "31d6cfe0d16ae931b73c59d7e0c089c0", "");
62          doTest(md4, "bde52cb31de33e46245e05fbdbd6fb24", "a");
63          doTest(md4, "a448017aaf21d8525fc10ae87aa6729d", "abc");
64          doTest(md4, "d9130a8164549fe818874806e1c7014b", "message digest");
65          doTest(md4, "d79e1c308aa5bbcdeea8ed63df412da9", "abcdefghijklmnopqrstuvwxyz");
66          doTest(md4, "043f8582f241db351ce627e153e7f0e4",
67                  "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789");
68          doTest(md4, "e33b4ddc9c38f2199c3e7b164fcc0536",
69                  "12345678901234567890123456789012345678901234567890123456789012345678901234567890");
70      }
71  
72      /**
73       * Original test vector found on <a href="http://en.wikipedia.org/wiki/MD4">wikipedia(en)</a>
74       * and <a href="http://fr.wikipedia.org/wiki/MD4">wikipedia(fr)</a>
75       */
76      @Test
77      public void testWikipediaVectors() throws NoSuchAlgorithmException, NoSuchProviderException {
78          MessageDigest md4 = MessageDigest.getInstance("MD4", MD4Provider.PROVIDER_NAME);
79          doTest(md4, "b94e66e0817dd34dc7858a0c131d4079", "Wikipedia, l'encyclopedie libre et gratuite");
80          doTest(md4, "1bee69a46ba811185c194762abaeae90", "The quick brown fox jumps over the lazy dog");
81          doTest(md4, "b86e130ce7028da59e672d56ad0113df", "The quick brown fox jumps over the lazy cog");
82      }
83  
84      /**
85       * Performs md4 digesting on the provided test vector and verifies that the
86       * result equals to the expected result.
87       * 
88       * @param md4 the md4 message digester
89       * @param expected the expected hex formatted string
90       * @param testVector the string message
91       */
92      private static void doTest(MessageDigest md4, String expected, String testVector) {
93          String result = asHex(md4.digest(testVector.getBytes()));
94          assertEquals(expected, result);
95      }
