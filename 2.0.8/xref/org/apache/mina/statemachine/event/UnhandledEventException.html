20  package org.apache.mina.statemachine.event;
21  
22  /**
23   * Thrown when an {@link Event} passed to a state machine couldn't be handled.
24   *
25   * @author <a href="http://mina.apache.org">Apache MINA Project</a>
26   */
27  public class UnhandledEventException extends RuntimeException {
28      private static final long serialVersionUID = -717373229954175430L;
29  
30      private final Event event;
31  
32      public UnhandledEventException(Event event) {
33          super("Unhandled event: " + event);
34          this.event = event;
35      }
36  
37      /**
38       * Returns the {@link Event} which couldn't be handled.
39       * 
40       * @return the {@link Event}.
41       */
42      public Event getEvent() {
43          return event;
44      }
45  }
