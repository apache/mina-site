020package org.apache.mina.example.tcp.perf;
021
022import java.io.IOException;
023import java.net.InetSocketAddress;
024import java.util.concurrent.atomic.AtomicInteger;
025
026import org.apache.mina.core.service.IoHandlerAdapter;
027import org.apache.mina.core.session.IdleStatus;
028import org.apache.mina.core.session.IoSession;
029import org.apache.mina.transport.socket.SocketSessionConfig;
030import org.apache.mina.transport.socket.nio.NioSocketAcceptor;
031
032/**
033 * An TCP server used for performance tests.
034 * 
035 * It does nothing fancy, except receiving the messages, and counting the number of
036 * received messages.
037 * 
038 * @author <a href="http://mina.apache.org">Apache MINA Project</a>
039 */
040public class TcpServer extends IoHandlerAdapter {
041    /** The listening port (check that it's not already in use) */
042    public static final int PORT = 18567;
043
044    /** The number of message to receive */
045    public static final int MAX_RECEIVED = 100000;
046
047    /** The starting point, set when we receive the first message */
048    private static long t0;
049
050    /** A counter incremented for every recieved message */
051    private AtomicInteger nbReceived = new AtomicInteger(0);
052
053    /**
054     * {@inheritDoc}
055     */
056    @Override
057    public void exceptionCaught(IoSession session, Throwable cause) throws Exception {
058        cause.printStackTrace();
059        session.close(true);
060    }
061
062    /**
063     * {@inheritDoc}
064     */
065    @Override
066    public void messageReceived(IoSession session, Object message) throws Exception {
067
068        int nb = nbReceived.incrementAndGet();
069
070        if (nb == 1) {
071            t0 = System.currentTimeMillis();
072        }
073
074        if (nb == MAX_RECEIVED) {
075            long t1 = System.currentTimeMillis();
076            System.out.println("-------------> end " + (t1 - t0));
077        }
078
079        if (nb % 10000 == 0) {
080            System.out.println("Received " + nb + " messages");
081        }
082
083        // If we want to test the write operation, uncomment this line
084        session.write(message);
085    }
086
087    /**
088     * {@inheritDoc}
089     */
090    @Override
091    public void sessionClosed(IoSession session) throws Exception {
092        System.out.println("Session closed...");
093
094        // Reinitialize the counter and expose the number of received messages
095        System.out.println("Nb message received : " + nbReceived.get());
096        nbReceived.set(0);
097    }
098
099    /**
100     * {@inheritDoc}
101     */
102    @Override
103    public void sessionCreated(IoSession session) throws Exception {
104        System.out.println("Session created...");
105    }
106
107    /**
108     * {@inheritDoc}
109     */
110    @Override
111    public void sessionIdle(IoSession session, IdleStatus status) throws Exception {
112        System.out.println("Session idle...");
113    }
114
115    /**
116     * {@inheritDoc}
117     */
118    @Override
119    public void sessionOpened(IoSession session) throws Exception {
120        System.out.println("Session Opened...");
121    }
122
123    /**
124     * Create the TCP server
125     */
126    public TcpServer() throws IOException {
127        NioSocketAcceptor acceptor = new NioSocketAcceptor();
128        acceptor.setHandler(this);
129
130        // The logger, if needed. Commented atm
131        //DefaultIoFilterChainBuilder chain = acceptor.getFilterChain();
132        //chain.addLast("logger", new LoggingFilter());
133
134        acceptor.bind(new InetSocketAddress(PORT));
135
136        System.out.println("Server started...");
137    }
138
139    /**
140     * The entry point.
141     */
142    public static void main(String[] args) throws IOException {
143        new TcpServer();
144    }
145}
