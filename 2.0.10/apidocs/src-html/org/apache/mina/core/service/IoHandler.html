020package org.apache.mina.core.service;
021
022import java.io.IOException;
023
024import org.apache.mina.core.session.IdleStatus;
025import org.apache.mina.core.session.IoSession;
026
027/**
028 * Handles all I/O events fired by MINA.
029 *
030 * @author <a href="http://mina.apache.org">Apache MINA Project</a>
031 *
032 * @see IoHandlerAdapter
033 */
034public interface IoHandler {
035    /**
036     * Invoked from an I/O processor thread when a new connection has been created.
037     * Because this method is supposed to be called from the same thread that
038     * handles I/O of multiple sessions, please implement this method to perform
039     * tasks that consumes minimal amount of time such as socket parameter
040     * and user-defined session attribute initialization.
041     */
042    void sessionCreated(IoSession session) throws Exception;
043
044    /**
045     * Invoked when a connection has been opened.  This method is invoked after
046     * {@link #sessionCreated(IoSession)}.  The biggest difference from
047     * {@link #sessionCreated(IoSession)} is that it's invoked from other thread
048     * than an I/O processor thread once thread model is configured properly.
049     */
050    void sessionOpened(IoSession session) throws Exception;
051
052    /**
053     * Invoked when a connection is closed.
054     */
055    void sessionClosed(IoSession session) throws Exception;
056
057    /**
058     * Invoked with the related {@link IdleStatus} when a connection becomes idle.
059     * This method is not invoked if the transport type is UDP; it's a known bug,
060     * and will be fixed in 2.0.
061     */
062    void sessionIdle(IoSession session, IdleStatus status) throws Exception;
063
064    /**
065     * Invoked when any exception is thrown by user {@link IoHandler}
066     * implementation or by MINA.  If <code>cause</code> is an instance of
067     * {@link IOException}, MINA will close the connection automatically.
068     */
069    void exceptionCaught(IoSession session, Throwable cause) throws Exception;
070
071    /**
072     * Invoked when a message is received.
073     */
074    void messageReceived(IoSession session, Object message) throws Exception;
075
076    /**
077     * Invoked when a message written by {@link IoSession#write(Object)} is
078     * sent out.
079     */
080    void messageSent(IoSession session, Object message) throws Exception;
081
082    /**
083     * Handle the closure of an half-duplex TCP channel
084     */
085    void inputClosed(IoSession session) throws Exception;
086}
