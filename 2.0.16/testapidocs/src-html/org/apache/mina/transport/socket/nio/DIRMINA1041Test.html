001package org.apache.mina.transport.socket.nio;
002
003import org.apache.mina.core.buffer.IoBuffer;
004import org.apache.mina.core.future.CloseFuture;
005import org.apache.mina.core.future.ConnectFuture;
006import org.apache.mina.core.future.WriteFuture;
007import org.apache.mina.core.service.IoHandlerAdapter;
008import org.apache.mina.core.session.IoSession;
009import org.apache.mina.filter.codec.ProtocolCodecFilter;
010import org.apache.mina.filter.codec.textline.TextLineCodecFactory;
011import org.apache.mina.transport.socket.SocketAcceptor;
012import org.apache.mina.transport.socket.SocketConnector;
013import org.apache.mina.util.AvailablePortFinder;
014import org.junit.After;
015import org.junit.Assert;
016import org.junit.Before;
017import org.junit.Test;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021import java.net.InetSocketAddress;
022import java.net.SocketAddress;
023
024public class DIRMINA1041Test {
025
026    private static final Logger LOG = LoggerFactory.getLogger(DIRMINA1041Test.class);
027    private static final String HOST = "localhost";
028    private static final int PORT = AvailablePortFinder.getNextAvailable(); 
029    private static final long TIMEOUT = 10000L;
030    private static int counter = 0;
031    private SocketAcceptor acceptor;
032    private SocketConnector connector;
033    
034    @Before
035    public void setUp() throws Exception {
036        acceptor = new NioSocketAcceptor();
037        acceptor.setReuseAddress(true);
038        acceptor.setHandler(new SomeAcceptHandler());
039        acceptor.bind(new InetSocketAddress(HOST, PORT));
040
041        connector = new NioSocketConnector();
042        connector.getSessionConfig().setReuseAddress(true);
043        connector.getFilterChain().addLast("codec", new ProtocolCodecFilter(new TextLineCodecFactory()));
044        connector.setHandler(new SomeConnectHandler());
045    }
046
047    @Test
048    public void testWrite() throws InterruptedException {
049        SocketAddress address = new InetSocketAddress(HOST, PORT);
050        
051        try {
052            for (int i = 0; i < 10000; i++) {
053                ConnectFuture future = connector.connect( address);
054    
055                if (!future.awaitUninterruptibly(TIMEOUT)) {
056                    
057                    Assert.fail("ConnectFuture did not complete.");
058                }
059                
060                IoSession session = future.getSession();
061                
062                if ( i % 1000 == 0 ) {
063                    System.out.println("Loop " + i +", counter = " + counter);
064                }
065    
066                WriteFuture writeFuture = session.write("Test" + i);
067                
068                //LOG.info("Waiting for WriteFuture to complete. Session: " + session);
069                if (!writeFuture.await(TIMEOUT)) {
070                    LOG.info("WriteFuture did not complete. Session: " + session);
071                    Assert.fail("WriteFuture did not complete. Session: " + session);
072                }
073                
074                CloseFuture closeFuture = session.closeOnFlush();
075                
076                if (!closeFuture.awaitUninterruptibly(TIMEOUT)) {
077                    Assert.fail("CloseFuture did not complete.");
078                }
079                
080                //Thread.sleep( 2 );
081            }
082        } catch (Exception e) {
083            e.printStackTrace();
084        }
085        
086        System.out.println("Done " + 100000  + " loops, counter = " + counter);
087    }
088
089    @After
090    public void tearDown() throws Exception {
091        try { connector.dispose(true); } catch (Throwable e) { e.printStackTrace(); }
092        try { acceptor.unbind(); acceptor.dispose(true); } catch (Throwable e) { e.printStackTrace(); }
093    }
094
095    private IoSession getSession() {
096        ConnectFuture future = connector.connect(new InetSocketAddress(HOST, PORT));
097        if (!future.awaitUninterruptibly(TIMEOUT)) {
098            
099            Assert.fail("ConnectFuture did not complete.");
100        }
101        return future.getSession();
102    }
103
104    private void closeSession(IoSession session) {
105        CloseFuture closeFuture = session.closeNow();
106        if (!closeFuture.awaitUninterruptibly(TIMEOUT)) {
107            Assert.fail("CloseFuture did not complete.");
108        }
109    }
110
111    private class SomeConnectHandler extends IoHandlerAdapter {
112        @Override
113        public void sessionClosed(IoSession session) throws Exception {
114            //LOG.info("Connector - Session closed : " + session);
115        }
116        
117        public void messageSent(IoSession session, Object message) throws Exception {
118            //LOG.info("message sent : " + message);
119        }
120    }
121
122    private class SomeAcceptHandler extends IoHandlerAdapter {
123        @Override
124        public void messageReceived(IoSession session, Object message) throws Exception {
125            //LOG.info("Message received : " + ((IoBuffer)message).toString() );
126            counter++;
127            //session.closeNow();
128        }
129    }
130}
