020package org.apache.mina.handler.demux;
021
022import org.apache.mina.core.session.IoSession;
023
024/**
025 * A handler interface that {@link DemuxingIoHandler} forwards
026 * <code>exceptionCaught</code> events to.  You have to register your
027 * handler with the type of exception you want to get notified using
028 * {@link DemuxingIoHandler#addExceptionHandler(Class, ExceptionHandler)}.
029 *
030 * @author <a href="http://mina.apache.org">Apache MINA Project</a>
031 */
032public interface ExceptionHandler<E extends Throwable> {
033    /**
034     * A {@link ExceptionHandler} that does nothing.  This is useful when
035     * you want to ignore an exception of a specific type silently.
036     */
037    ExceptionHandler<Throwable> NOOP = new ExceptionHandler<Throwable>() {
038        public void exceptionCaught(IoSession session, Throwable cause) {
039            // Do nothing
040        }
041    };
042
043    /**
044     * A {@link ExceptionHandler} that closes the session immediately.
045     * This is useful when you want to close the session when an exception of
046     * a specific type is raised.
047     */
048    ExceptionHandler<Throwable> CLOSE = new ExceptionHandler<Throwable>() {
049        public void exceptionCaught(IoSession session, Throwable cause) {
050            session.close(true);
051        }
052    };
053
054    /**
055     * Invoked when the specific type of exception is caught from the
056     * specified <code>session</code>.
057     * 
058     * @param session The current session
059     * @param cause the exception's cause
060     * @throws Exception If we can't process the event 
061     */
062    void exceptionCaught(IoSession session, E cause) throws Exception;
063}
