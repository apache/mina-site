020package org.apache.mina.filter.codec.statemachine;
021
022import org.apache.mina.core.buffer.IoBuffer;
023import org.apache.mina.filter.codec.ProtocolDecoderException;
024import org.apache.mina.filter.codec.ProtocolDecoderOutput;
025
026/**
027 * {@link DecodingState} which decodes <code>int</code> values in big-endian 
028 * order (high bytes come first).
029 * 
030 * @author <a href="http://mina.apache.org">Apache MINA Project</a>
031 */
032public abstract class IntegerDecodingState implements DecodingState {
033    private int counter;
034
035    /**
036     * {@inheritDoc}
037     */
038    public DecodingState decode(IoBuffer in, ProtocolDecoderOutput out) throws Exception {
039        int firstByte = 0;
040        int secondByte = 0;
041        int thirdByte = 0;
042        
043        while (in.hasRemaining()) {
044            switch (counter) {
045                case 0:
046                    firstByte = in.getUnsigned();
047                    break;
048                    
049                case 1:
050                    secondByte = in.getUnsigned();
051                    break;
052                    
053                case 2:
054                    thirdByte = in.getUnsigned();
055                    break;
056                    
057                case 3:
058                    counter = 0;
059                    return finishDecode((firstByte << 24) | (secondByte << 16) | (thirdByte << 8) | in.getUnsigned(), out);
060                    
061                default:
062                    throw new InternalError();
063            }
064            
065            counter++;
066        }
067
068        return this;
069    }
070
071    /**
072     * {@inheritDoc}
073     */
074    public DecodingState finishDecode(ProtocolDecoderOutput out) throws Exception {
075        throw new ProtocolDecoderException("Unexpected end of session while waiting for an integer.");
076    }
077
078    /**
079     * Invoked when this state has consumed a complete <code>int</code>.
080     * 
081     * @param value the integer.
082     * @param out the current {@link ProtocolDecoderOutput} used to write 
083     *        decoded messages.
084     * @return the next state if a state transition was triggered (use 
085     *         <code>this</code> for loop transitions) or <code>null</code> if 
086     *         the state machine has reached its end.
087     * @throws Exception if the read data violated protocol specification.
088     */
089    protected abstract DecodingState finishDecode(int value, ProtocolDecoderOutput out) throws Exception;
090}
