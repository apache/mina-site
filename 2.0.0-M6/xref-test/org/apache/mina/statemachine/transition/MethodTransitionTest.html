36  public class MethodTransitionTest extends RMockTestCase {
37      State currentState;
38      State nextState;
39      TestStateContext context;
40      Target target;
41      Method subsetAllArgsMethod1;
42      Method subsetAllArgsMethod2;
43      Event noArgsEvent;
44      Event argsEvent;
45      Object[] args;
46      
47      protected void setUp() throws Exception {
48          super.setUp();
49          
50          currentState = new State( "current" );
51          nextState = new State( "next" );
52          target = (Target) mock(Target.class);
53          subsetAllArgsMethod1 = Target.class.getMethod("subsetAllArgs", new Class[] {
54                  TestStateContext.class, B.class, A.class, Integer.TYPE
55          });
56          subsetAllArgsMethod2 = Target.class.getMethod("subsetAllArgs", new Class[] {
57                  Event.class, B.class, B.class, Boolean.TYPE
58          });
59          
60          args = new Object[] { new A(), new B(), new C(), new Integer(627438), Boolean.TRUE };
61          context = (TestStateContext) mock(TestStateContext.class);
62          noArgsEvent = new Event("event", context, new Object[0]);
63          argsEvent = new Event("event", context, args);
64      }
65  
66      public void testExecuteWrongEventId() throws Exception {
67          startVerification();
68          MethodTransition t = new MethodTransition("otherEvent", nextState, "noArgs", target);
69          assertFalse(t.execute(noArgsEvent));
70      }
71      
72      public void testExecuteNoArgsMethodOnNoArgsEvent() throws Exception {
73          target.noArgs();
74          startVerification();
75          MethodTransition t = new MethodTransition("event", nextState, "noArgs", target);
76          assertTrue(t.execute(noArgsEvent));
77      }
78      
79      public void testExecuteNoArgsMethodOnArgsEvent() throws Exception {
80          target.noArgs();
81          startVerification();

83          assertTrue(t.execute(argsEvent));
84      }
