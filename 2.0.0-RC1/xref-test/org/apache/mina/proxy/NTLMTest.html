90      /**
91       * Tests creating a type 1 message.
92       */
93      public void testType1Message() {
94          int customFlags = NTLMUtilities.FLAG_NEGOTIATE_UNICODE
95                  | NTLMUtilities.FLAG_NEGOTIATE_OEM
96                  | NTLMUtilities.FLAG_NEGOTIATE_NTLM
97                  | NTLMUtilities.FLAG_REQUEST_SERVER_AUTH_REALM
98                  | NTLMUtilities.FLAG_NEGOTIATE_DOMAIN_SUPPLIED
99                  | NTLMUtilities.FLAG_NEGOTIATE_WORKSTATION_SUPPLIED;
100 
101         byte[] osVer = new byte[8];
102         NTLMUtilities
103                 .writeOSVersion((byte) 5, (byte) 0, (short) 2195, osVer, 0);
104 
105         String msgType1 = asHex(NTLMUtilities.createType1Message("WORKSTATION",
106                 "DOMAIN", customFlags, osVer));
107         assertEquals(
108                 "4e544c4d53535000010000000732000006000600330000000b000b0028000000"
109                         + "050093080000000f574f524b53544154494f4e444f4d41494e",
110                 msgType1);
111 
112         assertEquals("050093080000000f", asHex(osVer));
113         
114         //Microsoft Windows XP [version 5.1.2600]
115         String os = System.getProperty("os.name");
116         if (os != null && os.toUpperCase().contains("WINDOWS") && 
117                 "5.1".equals(System.getProperty("os.version"))) {
118             String hex = asHex(NTLMUtilities.getOsVersion());
119             assertEquals("0501", hex.substring(0, 4));
120             assertEquals(16, hex.length());
121         }
122     }

128     public void testType3Message() throws Exception {
129         try {
130             MessageDigest.getInstance("MD4");
131         } catch (NoSuchAlgorithmException ex) {
132             logger.warn("No MD4 digest provider found !");
133             return;
134         }
135 
136         int flags = 0x00000001 | 0x00000200 | 0x00010000 | 0x00800000;
137         String msg = "4e544c4d53535000020000000c000c003000000001028100"
138                 + "0123456789abcdef0000000000000000620062003c000000"
139                 + "44004f004d00410049004e0002000c0044004f004d004100"
140                 + "49004e0001000c0053004500520056004500520004001400"
141                 + "64006f006d00610069006e002e0063006f006d0003002200"
142                 + "7300650072007600650072002e0064006f006d0061006900"
143                 + "6e002e0063006f006d0000000000";
144 
145         byte[] challengePacket = ByteUtilities.asByteArray(msg);
146         int serverFlags = NTLMUtilities
147                 .extractFlagsFromType2Message(challengePacket);
148         assertEquals(flags, serverFlags);
149 
150         NTLMUtilities
151                 .printTargetInformationBlockFromType2Message(challengePacket,
152                         serverFlags, new PrintWriter(System.out, true));
153 
154         byte[] osVer = new byte[8];
155         NTLMUtilities
156                 .writeOSVersion((byte) 5, (byte) 0, (short) 2195, osVer, 0);
157 
158         byte[] challenge = NTLMUtilities
159                 .extractChallengeFromType2Message(challengePacket);
160         assertEquals("0123456789abcdef", asHex(challenge));
161 
162         String expectedTargetInfoBlock = "02000c0044004f004d00410049004e00"
163                 + "01000c00530045005200560045005200"
164                 + "0400140064006f006d00610069006e00"
165                 + "2e0063006f006d000300220073006500"
166                 + "72007600650072002e0064006f006d00"
167                 + "610069006e002e0063006f006d000000" + "0000";
168 
169         byte[] targetInfo = NTLMUtilities.extractTargetInfoFromType2Message(
170                 challengePacket, null);
171         assertEquals(expectedTargetInfoBlock, asHex(targetInfo));
172 
173         assertEquals("DOMAIN", NTLMUtilities.extractTargetNameFromType2Message(
174                 challengePacket, new Integer(serverFlags)));
175 
176         serverFlags = 0x00000001 | 0x00000200;
177         String msgType3 = asHex(NTLMUtilities.createType3Message("user",
178                 "SecREt01", challenge, "DOMAIN", "WORKSTATION", serverFlags,
179                 null));
180 
181         String expected = "4e544c4d5353500003000000180018006a00000018001800"
182                 + "820000000c000c0040000000080008004c00000016001600"
183                 + "54000000000000009a0000000102000044004f004d004100"
184                 + "49004e00750073006500720057004f0052004b0053005400"
185                 + "4100540049004f004e00c337cd5cbd44fc9782a667af6d42"
186                 + "7c6de67c20c2d3e77c5625a98c1c31e81847466b29b2df46"
187                 + "80f39958fb8c213a9cc6";
188         assertEquals(expected, msgType3);
189     }

191     /**
192      * Tests flags manipulations.
193      */
194     public void testFlags() {
195         int flags = NTLMUtilities.FLAG_NEGOTIATE_UNICODE
196                 | NTLMUtilities.FLAG_REQUEST_SERVER_AUTH_REALM
197                 | NTLMUtilities.FLAG_NEGOTIATE_NTLM
198                 | NTLMUtilities.FLAG_NEGOTIATE_ALWAYS_SIGN;
199 
200         int flags2 = NTLMUtilities.FLAG_NEGOTIATE_UNICODE
201                 | NTLMUtilities.FLAG_REQUEST_SERVER_AUTH_REALM
202                 | NTLMUtilities.FLAG_NEGOTIATE_NTLM;
203 
204         assertEquals(flags2, flags
205                 & (~NTLMUtilities.FLAG_NEGOTIATE_ALWAYS_SIGN));
206         assertEquals(flags2, flags2
207                 & (~NTLMUtilities.FLAG_NEGOTIATE_ALWAYS_SIGN));
208         assertEquals("05820000", asHex(ByteUtilities.writeInt(flags)));
209 
210         byte[] testFlags = ByteUtilities.asByteArray("7F808182");
211         assertEquals("7f808182", asHex(testFlags));
212         ByteUtilities.changeByteEndianess(testFlags, 0, 4);
213         assertEquals("807f8281", asHex(testFlags));
214         ByteUtilities.changeByteEndianess(testFlags, 0, 4);
215         ByteUtilities.changeWordEndianess(testFlags, 0, 4);
216         assertEquals("8281807f", asHex(testFlags));
217     }
