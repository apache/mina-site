63          chain.addAfter( "B", "G", new EventOrderTestFilter( 'A' ) );
64          chain.addAfter( "D", "H", new EventOrderTestFilter( 'A' ) );
65          
66          String actual = "";
67          for( Iterator i = chain.getAll().iterator(); i.hasNext(); )

69              Entry e = ( Entry ) i.next();

179     private void run( String expectedResult )
180     {
181         chain.sessionCreated( session );
182         chain.sessionOpened( session );
183         chain.messageReceived( session, new Object() );
184         chain.filterWrite( session, new WriteRequest( new Object() ) );
185         chain.sessionIdle( session, IdleStatus.READER_IDLE );
186         chain.exceptionCaught( session, new Exception() );
187         chain.sessionClosed( session );
188         
189         result = formatResult( result );
190         expectedResult = formatResult( expectedResult );
191         
192         System.out.println( "Expected: " + expectedResult );
193         System.out.println( "Actual:   " + result );
194         Assert.assertEquals( expectedResult, result );
195     }
196     
197     private String formatResult( String result )
198     {
199         result = result.replaceAll( "//s", "" );
200         StringBuffer buf = new StringBuffer( result.length() * 4 / 3 );
201         for( int i = 0; i < result.length(); i++ )
202         {
203             buf.append( result.charAt( i ) );
204             if( i % 3 == 2 )
205             {
206                 buf.append(' ');
207             }
208         }
209         
210         return buf.toString();
211     }
