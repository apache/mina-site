164     public void testGetString() throws Exception
165     {
166         ByteBuffer buf = ByteBuffer.allocate( 16 );
167         CharsetDecoder decoder;
168 
169         decoder = Charset.forName( "ISO-8859-1" ).newDecoder();
170         buf.put( (byte) 'A' );
171         buf.put( (byte) 'B' );
172         buf.put( (byte) 'C' );
173         buf.put( (byte) 0 );
174         
175         buf.position( 0 );
176         Assert.assertEquals( "ABC", buf.getString( decoder ) );
177         Assert.assertEquals( 4, buf.position() );
178         
179         buf.position( 0 );
180         buf.limit( 1 );
181         Assert.assertEquals( "A", buf.getString( decoder ) );
182         Assert.assertEquals( 1, buf.position() );
183         
184         buf.clear();
185         Assert.assertEquals( "ABC", buf.getString( 10, decoder ) );
186         Assert.assertEquals( 10, buf.position() );
187         
188         buf.clear();
189         Assert.assertEquals( "A", buf.getString( 1, decoder ) );
190         Assert.assertEquals( 1, buf.position() );
191         
192         buf.clear();
193         buf.fillAndReset( buf.limit() );
194         decoder = Charset.forName( "UTF-16" ).newDecoder();
195         buf.put( (byte) 0 );
196         buf.put( (byte) 'A' );
197         buf.put( (byte) 0 );
198         buf.put( (byte) 'B' );
199         buf.put( (byte) 0 );
200         buf.put( (byte) 'C' );
201         buf.put( (byte) 0 );
202         buf.put( (byte) 0 );
203         
204         buf.position( 0 );
205         Assert.assertEquals( "ABC", buf.getString( decoder ) );
206         Assert.assertEquals( 8, buf.position() );
207 
208         buf.position( 0 );
209         buf.limit( 2 );
210         Assert.assertEquals( "A", buf.getString( decoder ) );
211         Assert.assertEquals( 2, buf.position() );
212

215         Assert.assertEquals( "A", buf.getString( decoder ) );
216         Assert.assertEquals( 2, buf.position() );

258     public void testPutString() throws Exception
259     {
260         CharsetEncoder encoder;
261         ByteBuffer buf = ByteBuffer.allocate( 16 );
262         encoder = Charset.forName( "ISO-8859-1" ).newEncoder();
263         
264         buf.putString( "ABC", encoder );
265         Assert.assertEquals( 3, buf.position() );
266         buf.clear();
267         Assert.assertEquals( 'A', buf.get( 0 ) );
268         Assert.assertEquals( 'B', buf.get( 1 ) );
269         Assert.assertEquals( 'C', buf.get( 2 ) );
270         
271         buf.putString( "D", 5, encoder );
272         Assert.assertEquals( 5, buf.position() );
273         buf.clear();
274         Assert.assertEquals( 'D', buf.get( 0 ) );
275         Assert.assertEquals( 0, buf.get( 1 ) );
276         
277         buf.putString( "EFG", 2, encoder );
278         Assert.assertEquals( 2, buf.position() );
279         buf.clear();
280         Assert.assertEquals( 'E', buf.get( 0 ) );
281         Assert.assertEquals( 'F', buf.get( 1 ) );
282         Assert.assertEquals( 'C', buf.get( 2 ) ); // C may not be overwritten
283 
284         // UTF-16: We specify byte order to omit BOM.
285         encoder = Charset.forName( "UTF-16BE" ).newEncoder();
286         buf.clear();
287         
288         buf.putString( "ABC", encoder );
289         Assert.assertEquals( 6, buf.position() );
290         buf.clear();
291         
292         Assert.assertEquals( 0, buf.get( 0 ) );
293         Assert.assertEquals( 'A', buf.get( 1 ) );
294         Assert.assertEquals( 0, buf.get( 2 ) );
295         Assert.assertEquals( 'B', buf.get( 3 ) );
296         Assert.assertEquals( 0, buf.get( 4 ) );
297         Assert.assertEquals( 'C', buf.get( 5 ) );
298         
299         buf.putString( "D", 10, encoder );
300         Assert.assertEquals( 10, buf.position() );
301         buf.clear();
302         Assert.assertEquals( 0, buf.get( 0 ) );
303         Assert.assertEquals( 'D', buf.get( 1 ) );
304         Assert.assertEquals( 0, buf.get( 2 ) );
305         Assert.assertEquals( 0, buf.get( 3 ) );
306         
307         buf.putString( "EFG", 4, encoder );
308         Assert.assertEquals( 4, buf.position() );
309         buf.clear();
310         Assert.assertEquals( 0, buf.get( 0 ) );
311         Assert.assertEquals( 'E', buf.get( 1 ) );
312         Assert.assertEquals( 0, buf.get( 2 ) );
313         Assert.assertEquals( 'F', buf.get( 3 ) );
314         Assert.assertEquals( 0, buf.get( 4 ) );   // C may not be overwritten
315         Assert.assertEquals( 'C', buf.get( 5 ) ); // C may not be overwritten
316         
317         // Test putting an emptry string
318         buf.putString( "", encoder );
319         Assert.assertEquals( 0, buf.position() );
320         buf.putString( "", 4, encoder );
321         Assert.assertEquals( 4, buf.position() );
322         Assert.assertEquals( 0, buf.get( 0 ) );
323         Assert.assertEquals( 0, buf.get( 1 ) );
324     }

340     public void testPutPrefixedString() throws Exception
341     {
342         CharsetEncoder encoder;
343         ByteBuffer buf = ByteBuffer.allocate( 16 );
344         buf.fillAndReset( buf.remaining() );
345         encoder = Charset.forName( "ISO-8859-1" ).newEncoder();
346         
347         // Without autoExpand
348         buf.putPrefixedString( "ABC", encoder );
349         Assert.assertEquals( 5, buf.position() );
350         Assert.assertEquals( 0, buf.get( 0 ) );
351         Assert.assertEquals( 3, buf.get( 1 ) );
352         Assert.assertEquals( 'A', buf.get( 2 ) );
353         Assert.assertEquals( 'B', buf.get( 3 ) );
354         Assert.assertEquals( 'C', buf.get( 4 ) );
355         
356         buf.clear();
357         try
358         {
359             buf.putPrefixedString( "123456789012345", encoder );
360             Assert.fail();
361         }
362         catch( BufferOverflowException e )
363         {
364             // OK
365         }
366         
367         // With autoExpand
368         buf.clear();
369         buf.setAutoExpand( true );
370         buf.putPrefixedString( "123456789012345", encoder );
371         Assert.assertEquals( 17, buf.position() );
372         Assert.assertEquals( 0, buf.get( 0 ) );
373         Assert.assertEquals( 15, buf.get( 1 ) );
374         Assert.assertEquals( '1', buf.get( 2 ) );
375         Assert.assertEquals( '2', buf.get( 3 ) );
376         Assert.assertEquals( '3', buf.get( 4 ) );
377         Assert.assertEquals( '4', buf.get( 5 ) );
378         Assert.assertEquals( '5', buf.get( 6 ) );
379         Assert.assertEquals( '6', buf.get( 7 ) );
380         Assert.assertEquals( '7', buf.get( 8 ) );
381         Assert.assertEquals( '8', buf.get( 9 ) );
382         Assert.assertEquals( '9', buf.get( 10 ) );
383         Assert.assertEquals( '0', buf.get( 11 ) );
384         Assert.assertEquals( '1', buf.get( 12 ) );
385         Assert.assertEquals( '2', buf.get( 13 ) );
386         Assert.assertEquals( '3', buf.get( 14 ) );
387         Assert.assertEquals( '4', buf.get( 15 ) );
388         Assert.assertEquals( '5', buf.get( 16 ) );
