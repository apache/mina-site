113     public void testCreateInstance() throws Exception
114     {
115         
116         DefaultIoFilterChainBuilder builder1 = new DefaultIoFilterChainBuilder();
117         builder1.addLast( "portFilter0", filter1 );
118         DefaultIoFilterChainBuilder builder2 = new DefaultIoFilterChainBuilder();
119         builder2.addLast( "portFilter0", filter1 );
120         builder2.addLast( "portFilter1", filter2 );
121         
122         /*
123          * Record expectations.
124          */
125         factory.createIoAcceptor();
126         mockFactory.setReturnValue( ioAcceptor );
127         factory.initIoSessionManager( ioAcceptor );
128         ioAcceptor.bind( new DummySocketAddress( ":110" ), popHandler, builder1 );
129         mockIoAcceptor.setMatcher( new IoAcceptorBindArgumentsMatcher() );
130         ioAcceptor.bind( new DummySocketAddress( "127.0.0.1:22" ), sshHandler, builder2 );
131         ioAcceptor.bind( new DummySocketAddress( "192.168.0.1:80" ),
132                 httpHandler, new DefaultIoFilterChainBuilder() );
133 
134         /*
135          * Replay.
136          */
137         mockIoAcceptor.replay();
138         mockFactory.replay();
139 
140         factory.setBindings( bindings );
141         Object o = factory.createInstance();
142 
143         /*
144          * Verify.
145          */
146         mockIoAcceptor.verify();
147         mockFactory.verify();
148 
149         assertSame( ioAcceptor, o );
150     }

228     public static class IoAcceptorBindArgumentsMatcher extends AbstractMatcher
229     {
230         protected boolean argumentMatches( Object expected, Object actual )
231         {
232             if( expected instanceof DefaultIoFilterChainBuilder && 
233                 actual instanceof DefaultIoFilterChainBuilder )
234             {
235                 DefaultIoFilterChainBuilder b1 = ( DefaultIoFilterChainBuilder ) expected;
236                 DefaultIoFilterChainBuilder b2 = ( DefaultIoFilterChainBuilder ) actual;
237                 
238                 List l1 = b1.getAll();
239                 List l2 = b2.getAll();
240                 if( l1.size() != l2.size() )
241                     return false;
242                 
243                 Iterator it1 = l1.iterator();
244                 Iterator it2 = l2.iterator();
245                 while( it1.hasNext() && it2.hasNext() )
246                 {
247                     Entry e1 = ( Entry ) it1.next();
248                     Entry e2 = ( Entry ) it2.next();
249                     
250                     if( !e1.getName().equals( e2.getName() ) )
251                         return false;                    
252                     if( e1.getFilter() != e2.getFilter() )
253                         return false;                    
254                 }
255                 
256                 return true;
257             }
258             return super.argumentMatches( expected, actual );
259         }
