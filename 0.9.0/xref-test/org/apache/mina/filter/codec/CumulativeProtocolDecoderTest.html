67      public void testCumulation() throws Exception
68      {
69          buf.put( (byte) 0 );
70          buf.flip();
71          
72          decoder.decode( session, buf, output );
73          Assert.assertEquals( 0, output.getValues().size() );
74          Assert.assertEquals( buf.limit(), buf.position() );
75          
76          buf.clear();
77          buf.put( (byte) 0 );
78          buf.put( (byte) 0 );
79          buf.put( (byte) 1 );
80          buf.flip();
81  
82          decoder.decode( session, buf, output );
83          Assert.assertEquals( 1, output.getValues().size() );
84          Assert.assertEquals( new Integer( 1 ), output.getValues().get( 0 ) );
85          Assert.assertEquals( buf.limit(), buf.position() );
86      }
87      
88      public void testRepeatitiveDecode() throws Exception
89      {
90          for( int i = 0; i < 4; i ++ )
91          {
92              buf.putInt( i );
93          }
94          buf.flip();
95          
96          decoder.decode( session, buf, output );
97          Assert.assertEquals( 4, output.getValues().size() );
98          Assert.assertEquals( buf.limit(), buf.position() );
99          
100         List expected = new ArrayList();
101         for( int i = 0; i < 4; i ++ )
102         {
103             expected.add( new Integer( i ) );
104         }
105         Assert.assertEquals( expected, output.getValues() );
106     }

120     private static class IntegerDecoder extends CumulativeProtocolDecoder
121     {
122         protected IntegerDecoder()
123         {
124             super( 4 );
125         }
126 
127         protected boolean doDecode( IoSession session, ByteBuffer in,
128                                     ProtocolDecoderOutput out ) throws Exception
129         {
130             Assert.assertTrue( in.hasRemaining() );
131             if( in.remaining() < 4 )
132                 return false;
133             
134             out.write( new Integer( in.getInt() ) );
135             return true;
136         }
137 
138         public void dispose() throws Exception
139         {
140         }
141         
142     }
