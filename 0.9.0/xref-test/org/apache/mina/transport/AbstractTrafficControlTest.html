74      protected abstract SocketAddress createServerSocketAddress( int port );

76      public void testSuspendResumeReadWrite() throws Exception
77      {
78          ConnectFuture future = connect( port, new ClientIoHandler() );
79          future.join();
80          IoSession session = future.getSession();
81          
82          write( session, "1" );
83          Thread.sleep( 250 );
84          assertEquals( "1", getReceived( session ) );
85          assertEquals( "1", getSent( session ) );
86          
87          session.suspendRead();
88          
89          write( session, "2" );
90          Thread.sleep( 250 );
91          assertEquals( "1", getReceived( session ) );
92          assertEquals( "12", getSent( session ) );
93          
94          session.suspendWrite();
95          
96          write( session, "3" );
97          Thread.sleep( 250 );
98          assertEquals( "1", getReceived( session ) );
99          assertEquals( "12", getSent( session ) );
100         
101         session.resumeRead();
102         
103         write( session, "4" );
104         Thread.sleep( 250 );
105         assertEquals( "12", getReceived( session ) );
106         assertEquals( "12", getSent( session ) );
107         
108         session.resumeWrite();
109         
110         write( session, "5" );
111         Thread.sleep( 250 );
112         assertEquals( "12345", getReceived( session ) );
113         assertEquals( "12345", getSent( session ) );
114         
115         session.suspendWrite();
116         
117         write( session, "6" );
118         Thread.sleep( 250 );
119         assertEquals( "12345", getReceived( session ) );
120         assertEquals( "12345", getSent( session ) );
121         
122         session.suspendRead();
123         session.resumeWrite();
124         
125         write( session, "7" );
126         Thread.sleep( 250 );
127         assertEquals( "12345", getReceived( session ) );
128         assertEquals( "1234567", getSent( session ) );
129         
130         session.resumeRead();
131         
132         Thread.sleep( 250 );
133         assertEquals( "1234567", getReceived( session ) );
134         assertEquals( "1234567", getSent( session ) );
135         
136         session.close().join();
137     }

139     private void write( IoSession session, String s ) throws Exception
140     {
141         session.write( ByteBuffer.wrap( s.getBytes( "ASCII" ) ) );
142     }

144     private static String getReceived( IoSession session ) 
145     {
146         return session.getAttribute( "received" ).toString();
147     }
148     
149     private static String getSent( IoSession session ) 
150     {
151         return session.getAttribute( "sent" ).toString();
152     }

154     public static class ClientIoHandler extends IoHandlerAdapter
155     {
156         public void sessionCreated( IoSession session ) throws Exception
157         {
158             super.sessionCreated( session );
159             session.setAttribute( "received", new StringBuffer() );
160             session.setAttribute( "sent", new StringBuffer() );
161         }
162 
163         public void messageReceived( IoSession session, Object message ) throws Exception
164         {
165             ByteBuffer buffer = ( ByteBuffer ) message;
166             byte[] data = new byte[ buffer.remaining() ];
167             buffer.get( data );
168             StringBuffer sb = ( StringBuffer ) session.getAttribute( "received" );
169             sb.append( new String( data, "ASCII" ) );
170         }
171 
172         public void messageSent( IoSession session, Object message ) throws Exception
173         {
174             ByteBuffer buffer = ( ByteBuffer ) message;
175             buffer.rewind();
176             byte[] data = new byte[ buffer.remaining() ];
177             buffer.get( data );
178             StringBuffer sb = ( StringBuffer ) session.getAttribute( "sent" );
179             sb.append( new String( data, "ASCII" ) );
180         }
181         
182     }
183     
184     public static class ServerIoHandler extends IoHandlerAdapter
185     {
186         public void messageReceived( IoSession session, Object message )
187                 throws Exception
188         {
189             // Just echo the received bytes.
190             ByteBuffer rb = ( ByteBuffer ) message;
191             ByteBuffer wb = ByteBuffer.allocate( rb.remaining() );
192             wb.put( rb );
193             wb.flip();
194             session.write( wb );                
195         }    
196     }
