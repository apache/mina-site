6   import java.io.IOException;

8   import java.util.HashMap;
9   import java.util.Map;

11  import org.apache.mina.common.IoFilterChainBuilder;
12  import org.apache.mina.common.IoHandler;
13  import org.apache.mina.common.support.BaseIoAcceptor;
14  import org.apache.mina.transport.vmpipe.support.VmPipe;

16  /***
17   * Binds the specified {@link IoHandler} to the specified
18   * {@link VmPipeAddress}.
19   * 
20   * @author The Apache Directory Project (dev@directory.apache.org)
21   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
22   */
23  public class VmPipeAcceptor extends BaseIoAcceptor
24  {
25      static final Map boundHandlers = new HashMap();
26  
27      public void bind( SocketAddress address, IoHandler handler, IoFilterChainBuilder filterChainBuilder ) throws IOException
28      {
29          if( address == null )
30              throw new NullPointerException( "address" );
31          if( handler == null )
32              throw new NullPointerException( "handler" );
33          if( !( address instanceof VmPipeAddress ) )
34              throw new IllegalArgumentException(
35                      "address must be VmPipeAddress." );
36  
37          if( filterChainBuilder == null )
38          {
39              filterChainBuilder = IoFilterChainBuilder.NOOP;
40          }
41  
42          synchronized( boundHandlers )
43          {
44              if( boundHandlers.containsKey( address ) )
45              {
46                  throw new IOException( "Address already bound: " + address );
47              }
48  
49              boundHandlers.put( address, 
50                                 new VmPipe( this,
51                                            ( VmPipeAddress ) address,
52                                            handler, filterChainBuilder ) );
53          }
54      }
55  
56      public void unbind( SocketAddress address )
57      {
58          // TODO: DIRMINA-93
59          if( address == null )
60              throw new NullPointerException( "address" );
61  
62          synchronized( boundHandlers )
63          {
64              boundHandlers.remove( address );
65          }
66      }
67  }
