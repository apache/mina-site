22      protected void doWrite( IoSession session, WriteRequest writeRequest )
23      {
24          DatagramSessionImpl s = ( DatagramSessionImpl ) session;
25          Queue writeRequestQueue = s.getWriteRequestQueue();
26          
27          synchronized( writeRequestQueue )
28          {
29              writeRequestQueue.push( writeRequest );
30              if( writeRequestQueue.size() == 1 && session.getTrafficMask().isWritable() )
31              {
32                  // Notify DatagramSessionManager only when writeRequestQueue was empty.
33                  s.getManagerDelegate().flushSession( s );
34              }
35          }
36      }

39      {

41          DatagramSessionManager manager = s.getManagerDelegate();
42          if( manager instanceof DatagramConnectorDelegate )
43          {
44              ( ( DatagramConnectorDelegate ) manager ).closeSession( s );
45          }
46          else
47          {
48              closeFuture.setClosed();
49          }
50      }
