58      /***
59       * Creates a new instance.
60       */
61      public SocketSessionImpl(
62              IoSessionManager manager,
63              SocketChannel ch, IoHandler defaultHandler )
64      {
65          this.manager = manager;
66          this.ioProcessor = SocketIoProcessor.getInstance();
67          this.filterChain = new SocketFilterChain( this );
68          this.ch = ch;
69          this.writeRequestQueue = new Queue();
70          this.handler = defaultHandler;
71          this.remoteAddress = ch.socket().getRemoteSocketAddress();
72          this.localAddress = ch.socket().getLocalSocketAddress();
73      }
