32  /***
33   * The default implementation of {@link IoFilterChainBuilder} which is useful
34   * in most cases.  {@link DefaultIoFilterChainBuilder} has an identical interface
35   * with {@link IoFilter}; it contains a list of {@link IoFilter}s that you can
36   * modify. The {@link IoFilter}s which are added to this builder will be appended
37   * to the {@link IoFilterChain} when {@link #buildFilterChain(IoFilterChain)} is
38   * invoked.
39   * <p>
40   * However, the identical interface doesn't mean that it behaves in an exactly
41   * same way with {@link IoFilterChain}.  {@link DefaultIoFilterChainBuilder}
42   * doesn't manage the life cycle of the {@link IoFilter}s at all, and the
43   * existing {@link IoSession}s won't get affected by the changes in this builder.
44   * {@link IoFilterChainBuilder}s affect only newly created {@link IoSession}s.
45   * 
46   * <pre>
47   * IoAcceptor acceptor = ...;
48   * DefaultIoFilterChainBuilder builder = acceptor.getFilterChain();
49   * builder.addLast( "myFilter", new MyFilter() );
50   * ...
51   * </pre>
52   *
53   * @author The Apache Directory Project (dev@directory.apache.org)
54   * @version $Rev: 350135 $, $Date: 2005-12-01 12:43:29 +0900 $
55   */
56  public class DefaultIoFilterChainBuilder implements IoFilterChainBuilder
57  {
58      private List entries = new ArrayList();
59      private final Map entriesByName = new HashMap();
60      
61      /***
62       * Creates a new instance with an empty filter list.
63       */
64      public DefaultIoFilterChainBuilder()
65      {
66      }
67  
68      /***
69       * @see IoFilterChain#getEntry(String)
70       */
71      public synchronized Entry getEntry( String name )
72      {
73          return ( Entry ) entriesByName.get( name );
74      }
75  
76      /***
77       * @see IoFilterChain#get(String)
78       */
79      public synchronized IoFilter get( String name )
80      {
81          Entry e = getEntry( name );
82          if( e == null )
83          {
84              return null;
85          }
86          
87          return e.getFilter();
88      }
89      
90      /***
91       * @see IoFilterChain#getAll()
92       */
93      public List getAll()
94      {
95          return new ArrayList( entries );
96      }
97      
98      /***
99       * @see IoFilterChain#getAllReversed()
100      */
101     public List getAllReversed()
102     {
103         List result = getAll();
104         Collections.reverse( result );
105         return result;
106     }
107 
108     /***
109      * @see IoFilterChain#contains(String)
110      */
111     public boolean contains( String name )
112     {
113         return getEntry( name ) != null;
114     }
115     
116     /***
117      * @see IoFilterChain#contains(IoFilter)
118      */
119     public boolean contains( IoFilter filter )
120     {
121         for( Iterator i = entries.iterator(); i.hasNext(); )
122         {
123             Entry e = ( Entry ) i.next();
124             if( e.getFilter() == filter )
125             {
126                 return true;
127             }
128         }
129         
130         return false;
131     }
132     
133     /***
134      * @see IoFilterChain#contains(Class)
135      */
136     public boolean contains( Class filterType )
137     {
138         for( Iterator i = entries.iterator(); i.hasNext(); )
139         {
140             Entry e = ( Entry ) i.next();
141             if( filterType.isAssignableFrom( e.getFilter().getClass() ) )
142             {
143                 return true;
144             }
145         }
146         
147         return false;
148     }
149     
150     /***
151      * @see IoFilterChain#addFirst(String, IoFilter)
152      */
153     public synchronized void addFirst( String name, IoFilter filter )
154     {
155         register( 0, new EntryImpl( name, filter ) );
156     }
157     
158     /***
159      * @see IoFilterChain#addLast(String, IoFilter)
160      */
161     public synchronized void addLast( String name, IoFilter filter )
162     {
163         register( entries.size(), new EntryImpl( name, filter ) );
164     }
165 
166     /***
167      * @see IoFilterChain#addBefore(String, String, IoFilter)
168      */
169     public synchronized void addBefore( String baseName, String name, IoFilter filter )
170     {
171         checkBaseName( baseName );
172         
173         for( ListIterator i = entries.listIterator(); i.hasNext(); )
174         {
175             Entry base = ( Entry ) i.next();
176             if( base.getName().equals( baseName ) )
177             {
178                 register( i.previousIndex(), new EntryImpl( name, filter ) );
179                 break;
180             }
181         }
182     }
183 
184     /***
185      * @see IoFilterChain#addAfter(String, String, IoFilter)
186      */
187     public synchronized void addAfter( String baseName, String name, IoFilter filter )
188     {
189         checkBaseName( baseName );
190         
191         List entries = new ArrayList( this.entries );
192         
193         for( ListIterator i = entries.listIterator(); i.hasNext(); )
194         {
195             Entry base = ( Entry ) i.next();
196             if( base.getName().equals( baseName ) )
197             {
198                 register( i.nextIndex(), new EntryImpl( name, filter ) );
199                 break;
200             }
201         }
202     }
203 
204     /***
205      * @see IoFilterChain#remove(String)
206      */
207     public synchronized IoFilter remove( String name )
208     {
209         if( name == null )
210         {
211             throw new NullPointerException( "name" );
212         }
213 
214         for( ListIterator i = entries.listIterator(); i.hasNext(); )
215         {
216             Entry e = ( Entry ) i.next();
217             if( e.getName().equals( name ) )
218             {
219                 deregister( i.previousIndex(), e );
220                 return e.getFilter();
221             }
222         }
223         
224         throw new IllegalArgumentException( "Unknown filter name: " + name );
225     }
226 
227     /***
228      * @see IoFilterChain#clear()
229      */
230     public synchronized void clear() throws Exception
231     {
232         entries = new ArrayList();
233         entriesByName.clear();
234     }
235     
236     public void buildFilterChain( IoFilterChain chain ) throws Exception
237     {
238         for( Iterator i = entries.iterator(); i.hasNext(); )
239         {
240             Entry e = ( Entry ) i.next();
241             chain.addLast( e.getName(), e.getFilter() );
242         }
243     }
244     
245     public String toString()
246     {
247         StringBuffer buf = new StringBuffer();
248         buf.append( "{ " );
249         
250         boolean empty = true;
251         
252         for( Iterator i = entries.iterator(); i.hasNext(); )
253         {
254             Entry e = ( Entry ) i.next();
255             if( !empty )
256             {
257                 buf.append( ", " );
258             }
259             else
260             {
261                 empty = false;
262             }
263             
264             buf.append( '(' );
265             buf.append( e.getName() );
266             buf.append( ':' );
267             buf.append( e.getFilter() );
268             buf.append( ')' );
269         }
270         
271         if( empty )
272         {
273             buf.append( "empty" );
274         }
275         
276         buf.append( " }" );
277         
278         return buf.toString();
279     }
280 
281     private void checkBaseName( String baseName )
282     {
283         if( baseName == null )
284         {
285             throw new NullPointerException( "baseName" );
286         }
287         if( !entriesByName.containsKey( baseName ) )
288         {
289             throw new IllegalArgumentException( "Unknown filter name: " + baseName );
290         }
291     }
292 
293     private void register( int index, Entry e )
294     {
295         if( entriesByName.containsKey( e.getName() ) )
296         {
297             throw new IllegalArgumentException( "Other filter is using the same name: " + e.getName() );
298         }
299 
300         List newEntries = new ArrayList( entries );
301         newEntries.add( index, e );
302         this.entries = newEntries;
303         entriesByName.put( e.getName(), e );
304     }
305     
306     private void deregister( int index, Entry e )
307     {
308         List newEntries = new ArrayList( entries );
309         newEntries.remove( index );
310         this.entries = newEntries;
311         entriesByName.remove( e.getName() );
312     }

314     private static class EntryImpl implements Entry
315     {
316         private final String name;
317         private final IoFilter filter;
318         
319         private EntryImpl( String name, IoFilter filter )
320         {
321             if( name == null )
322             {
323                 throw new NullPointerException( "name" );
324             }
325             if( filter == null )
326             {
327                 throw new NullPointerException( "filter" );
328             }
329             
330             this.name = name;
331             this.filter = filter;
332         }
333 
334         public String getName()
335         {
336             return name;
337         }
338         
339         public IoFilter getFilter()
340         {
341             return filter;
342         }
343 
344         public NextFilter getNextFilter()
345         {
346             throw new IllegalStateException();
347         }
348     }
