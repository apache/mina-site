19  package org.apache.mina.common;
20  
21  import java.nio.channels.SelectionKey;
22  
23  /***
24   * A type-safe mask that is used to control the traffic of {@link IoSession}
25   * with {@link IoSession#setTrafficMask(TrafficMask)}.
26   *
27   * @author The Apache Directory Project (dev@directory.apache.org)
28   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
29   */
30  public class TrafficMask
31  {
32      /***
33       * This mask suspends both reads and writes.
34       */
35      public static final TrafficMask NONE = new TrafficMask( 0, "none" );
36      /***
37       * This mask suspends writes, and resumes reads if reads were suspended.
38       */
39      public static final TrafficMask READ = new TrafficMask( SelectionKey.OP_READ, "read" );
40      /***
41       * This mask suspends reads, and resumes writes if writes were suspended.
42       */
43      public static final TrafficMask WRITE = new TrafficMask( SelectionKey.OP_WRITE, "write" );
44      /***
45       * This mask resumes both reads and writes if any of them were suspended.
46       */
47      public static final TrafficMask ALL =
48          new TrafficMask( SelectionKey.OP_READ | SelectionKey.OP_WRITE, "all" );
49      
50      /***
51       * Returns an appropriate {@link TrafficMask} instance from the
52       * specified <tt>interestOps</tt>.
53       * @see SelectionKey
54       */
55      public static TrafficMask getInstance( int interestOps )
56      {
57          boolean read = ( interestOps & SelectionKey.OP_READ ) != 0;
58          boolean write = ( interestOps & SelectionKey.OP_WRITE ) != 0;
59          if( read )
60          {
61              if( write )
62              {
63                  return ALL;
64              }
65              else
66              {
67                  return READ;
68              }
69          }
70          else if( write )
71          {
72              return WRITE;
73          }
74          else
75          {
76              return NONE;
77          }
78      }
79      
80      private final int interestOps;
81      private final String name;
82      
83      private TrafficMask( int interestOps, String name )
84      {
85          this.interestOps = interestOps;
86          this.name = name;
87      }
88  
89      /***
90       * Returns the name of this mask.
91       */
92      public String getName()
93      {
94          return name;
95      }
96      
97      /***
98       * Returns <tt>true</tt> if this mask allows a read operation.
99       */
100     public boolean isReadable()
101     {
102         return ( interestOps & SelectionKey.OP_READ ) != 0;
103     }
104     
105     /***
106      * Returns <tt>true</tt> if this mask allows a write operation.
107      */
108     public boolean isWritable()
109     {
110         return ( interestOps & SelectionKey.OP_WRITE ) != 0;
111     }
112     
113     /***
114      * Returns an interestOps of {@link SelectionKey} for this mask.
115      */
116     public int getInterestOps()
117     {
118         return interestOps;
119     }
120 
121     /***
122      * Peforms an <tt>AND</tt> operation on this mask with the specified
123      * <tt>mask</tt> and returns the result.
124      */
125     public TrafficMask and( TrafficMask mask )
126     {
127         return getInstance( interestOps & mask.interestOps );
128     }
129     
130     /***
131      * Peforms an <tt>OR</tt> operation on this mask with the specified
132      * <tt>mask</tt> and returns the result.
133      */
134     public TrafficMask or( TrafficMask mask )
135     {
136         return getInstance( interestOps | mask.interestOps );
137     }
138     
139     /***
140      * Returns a negated mask of this one.
141      */
142     public TrafficMask not()
143     {
144         return getInstance( ~interestOps );
145     }
146     
147     /***
148      * Peforms an <tt>XOR</tt> operation on this mask with the specified
149      * <tt>mask</tt> and returns the result.
150      */
151     public TrafficMask xor( TrafficMask mask )
152     {
153         return getInstance( interestOps ^ mask.interestOps );
154     }
155     
156     public String toString()
157     {
158         return name;
159     }
160 }
