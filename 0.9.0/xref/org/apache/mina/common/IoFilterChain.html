19  package org.apache.mina.common;
20  
21  import java.util.List;
22  
23  import org.apache.mina.common.IoFilter.NextFilter;
24  
25  /***
26   * A container of {@link IoFilter}s that forwards {@link IoHandler} events
27   * to the consisting filters and terminal {@link IoHandler} sequentially.
28   * Every {@link IoSession} has its own {@link IoFilterChain} (1-to-1 relationship). 
29   * 
30   * @author The Apache Directory Project (dev@directory.apache.org)
31   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
32   */
33  public interface IoFilterChain {
34      /***
35       * Returns the parent {@link IoSession} of this chain.
36       * @return {@link IoSession}
37       */
38      IoSession getSession();
39      
40      /***
41       * Returns the {@link Entry} with the specified <tt>name</tt> in this chain.
42       * @return <tt>null</tt> if there's no such name in this chain
43       */
44      Entry getEntry( String name );
45      
46      /***
47       * Returns the {@link IoFilter} with the specified <tt>name</tt> in this chain.
48       * @return <tt>null</tt> if there's no such name in this chain
49       */
50      IoFilter get( String name );
51      
52      /***
53       * Returns the {@link NextFilter} of the {@link IoFilter} with the
54       * specified <tt>name</tt> in this chain.
55       * @return <tt>null</tt> if there's no such name in this chain
56       */
57      NextFilter getNextFilter( String name );
58      
59      /***
60       * Returns the list of all {@link Entry}s this chain contains.
61       */
62      List getAll();
63      
64      /***
65       * Returns the reversed list of all {@link Entry}s this chain contains.
66       */
67      List getAllReversed();
68      
69      /***
70       * Returns <tt>true</tt> if this chain contains an {@link IoFilter} with the
71       * specified <tt>name</tt>.
72       */
73      boolean contains( String name );
74      
75      /***
76       * Returns <tt>true</tt> if this chain contains the specified <tt>filter</tt>.
77       */
78      boolean contains( IoFilter filter );
79      
80      /***
81       * Returns <tt>true</tt> if this chain contains an {@link IoFilter} of the
82       * specified <tt>filterType</tt>.
83       */
84      boolean contains( Class filterType );
85      
86      /***
87       * Adds the specified filter with the specified name at the beginning of this chain.
88       * @throws IoFilterLifeCycleException
89       *             if {@link IoFilter#onPostAdd(IoFilterChain, String, NextFilter)} or
90       *             {@link IoFilter#init()} throws an exception.
91       */
92      void addFirst( String name, IoFilter filter );
93  
94      /***
95       * Adds the specified filter with the specified name at the end of this chain.
96       * @throws IoFilterLifeCycleException
97       *             if {@link IoFilter#onPostAdd(IoFilterChain, String, NextFilter)} or
98       *             {@link IoFilter#init()} throws an exception.
99       */
100     void addLast( String name, IoFilter filter );
101 
102     /***
103      * Adds the specified filter with the specified name just before the filter whose name is
104      * <code>baseName</code> in this chain.
105      * @throws IoFilterLifeCycleException
106      *             if {@link IoFilter#onPostAdd(IoFilterChain, String, NextFilter)} or
107      *             {@link IoFilter#init()} throws an exception.
108      */
109     void addBefore( String baseName, String name, IoFilter filter );
110 
111     /***
112      * Adds the specified filter with the specified name just after the filter whose name is
113      * <code>baseName</code> in this chain.
114      * @throws IoFilterLifeCycleException
115      *             if {@link IoFilter#onPostAdd(IoFilterChain, String, NextFilter)} or
116      *             {@link IoFilter#init()} throws an exception.
117      */
118     void addAfter( String baseName, String name, IoFilter filter );
119 
120     /***
121      * Removes the filter with the specified name from this chain.
122      * @throws IoFilterLifeCycleException
123      *             if {@link IoFilter#onPostRemove(IoFilterChain, String, NextFilter)} or
124      *             {@link IoFilter#destroy()} throws an exception.
125      */
126     IoFilter remove( String name );
127 
128     /***
129      * Removes all filters added to this chain.
130      * @throws Exception if {@link IoFilter#onPostRemove(IoFilterChain, String, NextFilter)} thrown an exception.
131      */
132     void clear() throws Exception;
133     
134     /***
135      * Represents a name-filter pair that an {@link IoFilterChain} contains.
136      *
137      * @author The Apache Directory Project (dev@directory.apache.org)
138      * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
139      */
140     public interface Entry
141     {
142         /***
143          * Returns the name of the filter.
144          */
145         String getName();
146         
147         /***
148          * Returns the filter.
149          */
150         IoFilter getFilter();
151         
152         /***
153          * Returns the {@link NextFilter} of the filter.
154          * 
155          * @throws IllegalStateException if the {@link NextFilter} is not available
156          */
157         NextFilter getNextFilter();
158     }
159 }
