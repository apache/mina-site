45  /***
46   * A pooled byte buffer used by MINA applications.
47   * <p>
48   * This is a replacement for {@link java.nio.ByteBuffer}. Please refer to
49   * {@link java.nio.ByteBuffer} and {@link java.nio.Buffer} documentation for
50   * usage.  MINA does not use NIO {@link java.nio.ByteBuffer} directly for two
51   * reasons:
52   * <ul>
53   *   <li>It doesn't provide useful getters and putters such as
54   *       <code>fill</code>, <code>get/putString</code>, and
55   *       <code>get/putAsciiInt()</code> enough.</li>
56   *   <li>It is hard to distinguish if the buffer is created from MINA buffer
57   *       pool or not.  MINA have to return used buffers back to pool.</li>
58   *   <li>It is difficult to write variable-length data due to its fixed
59   *       capacity</li>
60   * </ul>
61   * 
62   * <h2>Allocation</h2>
63   * <p>
64   * You can get a heap buffer from buffer pool:
65   * <pre>
66   * ByteBuffer buf = ByteBuffer.allocate(1024, false);
67   * </pre>
68   * you can also get a direct buffer from buffer pool:
69   * <pre>
70   * ByteBuffer buf = ByteBuffer.allocate(1024, true);
71   * </pre>
72   * or you can let MINA choose:
73   * <pre>
74   * ByteBuffer buf = ByteBuffer.allocate(1024);
75   * </pre>
76   * 
77   * <h2>Acquire/Release</h2>
78   * <p>
79   * <b>Please note that you never need to release the allocated buffer</b>
80   * because MINA will release it automatically when:
81   * <ul>
82   *   <li>You pass the buffer by calling {@link IoSession#write(Object)}.</li>
83   *   <li>You pass the buffer by calling {@link IoFilter.NextFilter#filterWrite(IoSession,IoFilter.WriteRequest)}.</li>
84   *   <li>You pass the buffer by calling {@link ProtocolEncoderOutput#write(ByteBuffer)}.</li>
85   * </ul>
86   * And, you don't need to release any {@link ByteBuffer} which is passed as a parameter
87   * of {@link IoHandler#messageReceived(IoSession, Object)} method.  They are released
88   * automatically when the method returns.
89   * <p>
90   * You have to release buffers manually by calling {@link #release()} when:
91   * <ul>
92   *   <li>You allocated a buffer, but didn't pass the buffer to any of two methods above.</li>
93   *   <li>You called {@link #acquire()} to prevent the buffer from being released.</li>
94   * </ul>
95   * 
96   * <h2>Wrapping existing NIO buffers and arrays</h2>
97   * <p>
98   * This class provides a few <tt>wrap(...)</tt> methods that wraps
99   * any NIO buffers and byte arrays.  Wrapped MINA buffers are not returned
100  * to the buffer pool by default to prevent unexpected memory leakage by default.
101  * In case you want to make it pooled, you can call {@link #setPooled(boolean)}
102  * with <tt>true</tt> flag to enable pooling.
103  *
104  * <h2>AutoExpand</h2>
105  * <p>
106  * Writing variable-length data using NIO <tt>ByteBuffers</tt> is not really
107  * easy, and it is because its size is fixed.  MINA <tt>ByteBuffer</tt>
108  * introduces <tt>autoExpand</tt> property.  If <tt>autoExpand</tt> property
109  * is true, you never get {@link BufferOverflowException} or
110  * {@link IndexOutOfBoundsException} (except when index is negative).
111  * It automatically expands its capacity and limit value.  For example:
112  * <pre>
113  * String greeting = messageBundle.getMessage( "hello" );
114  * ByteBuffer buf = ByteBuffer.allocate( 16 );
115  * // Turn on autoExpand (it is off by default)
116  * buf.setAutoExpand( true );
117  * buf.putString( greeting, utf8encoder );
118  * </pre>
119  * NIO <tt>ByteBuffer</tt> is reallocated by MINA <tt>ByteBuffer</tt> behind
120  * the scene if the encoded data is larger than 16 bytes.  Its capacity will
121  * increase by two times, and its limit will increase to the last position
122  * the string is written.
123  * 
124  * @author The Apache Directory Project (dev@directory.apache.org)
125  * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
126  */

161     public static ByteBuffer allocate( int capacity )
162     {
163         try
164         {
165             // first try to allocate direct buffer
166             return allocate( capacity, true );
167         }
168         catch( OutOfMemoryError e )
169         {
170             // if failed, try heap
171             return allocate( capacity, false );
172         }
173     }

186         buf.init( nioBuffer );

188     }

190     private static DefaultByteBuffer allocateContainer()
191     {
192         DefaultByteBuffer buf;
193         synchronized( containerStack )
194         {
195             buf = ( DefaultByteBuffer ) containerStack.pop();
196         }
197         
198         if( buf == null )
199         {
200             buf = new DefaultByteBuffer();
201         }
202         return buf;
203     }
204     
205     private static java.nio.ByteBuffer allocate0( int capacity, boolean direct )
206     {
207         Stack[] bufferStacks = direct? directBufferStacks : heapBufferStacks;
208         int idx = getBufferStackIndex( bufferStacks, capacity );
209         Stack stack = bufferStacks[ idx ];
210 
211         java.nio.ByteBuffer buf;
212         synchronized( stack )
213         {
214             buf = ( java.nio.ByteBuffer ) stack.pop();
215         }
216 
217         if( buf == null )
218         {
219             buf = direct ? java.nio.ByteBuffer.allocateDirect( MINIMUM_CAPACITY << idx ) :
220                            java.nio.ByteBuffer.allocate( MINIMUM_CAPACITY << idx );
221         }
222         
223         return buf;
224     }
225     
226     private static void release0( java.nio.ByteBuffer buf )
227     {
228         Stack[] bufferStacks = buf.isDirect()? directBufferStacks : heapBufferStacks;
229         Stack stack = bufferStacks[ getBufferStackIndex( bufferStacks, buf.capacity() ) ];
230         synchronized( stack )
231         {
232             // push back
233             stack.push( buf );
234         }
235     }

267     private static int getBufferStackIndex( Stack[] bufferStacks, int size )
268     {
269         int targetSize = MINIMUM_CAPACITY;
270         int stackIdx = 0;
271         while( size > targetSize )
272         {
273             targetSize <<= 1;
274             stackIdx ++ ;
275             if( stackIdx >= bufferStacks.length )
276             {
277                 throw new IllegalArgumentException(
278                         "Buffer size is too big: " + size );
279             }
280         }
281 
282         return stackIdx;
283     }

547     public int hashCode()
548     {
549         int h = 1;
550         int p = position();
551         for( int i = limit() - 1; i >= p; i -- )
552         {
553             h = 31 * h + get( i );
554         }
555         return h;

558     public boolean equals( Object o )
559     {
560         if( !( o instanceof ByteBuffer ) )
561         {
562             return false;
563         }
564         
565         ByteBuffer that = ( ByteBuffer ) o;
566         if( this.remaining() != that.remaining() )
567         {
568             return false;
569         }
570         
571         int p = this.position();
572         for( int i = this.limit() - 1, j = that.limit() - 1; i >= p; i --, j --)
573         {
574             byte v1 = this.get( i );
575             byte v2 = that.get( j );
576             if( v1 != v2 )
577             {
578                 return false;
579             }
580         }
581         return true;
582     }

584     public int compareTo( Object o )
585     {
586         ByteBuffer that = ( ByteBuffer ) o;
587         int n = this.position() + Math.min( this.remaining(), that.remaining() );
588         for( int i = this.position(), j = that.position(); i < n; i ++, j ++ )
589         {
590             byte v1 = this.get( i );
591             byte v2 = that.get( j );
592             if( v1 == v2 )
593             {
594                 continue;
595             }
596             if( v1 < v2 )
597             {
598                 return -1;
599             }
600             
601             return +1;
602         }
603         return this.remaining() - that.remaining();
604     }
