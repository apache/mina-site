21  import java.net.SocketAddress;

24  /***
25   * A handle which represents connection between two endpoints regardless of 
26   * transport types.
27   * <p>
28   * {@link IoSession} provides user-defined attributes.  User-defined attributes
29   * are application-specific data which is associated with a session.
30   * It often contains objects that represents the state of a higher-level protocol
31   * and becomes a way to exchange data between filters and handlers.
32   * 
33   * <h3>Adjusting Transport Type Specific Properties</h3>
34   * <p>
35   * You can simply downcast the session to an appropriate subclass.
36   * </p>
37   * 
38   * <h3>Thread Safety</h3>
39   * <p>
40   * {@link IoSession} is thread-safe.  But please note that performing
41   * more than one {@link #write(Object)} calls at the same time will
42   * cause the {@link IoFilter#filterWrite(IoFilter.NextFilter, IoSession, IoFilter.WriteRequest)}
43   * is executed simnutaneously, and therefore you have to make sure the
44   * {@link IoFilter} implementations you're using are thread-safe, too. 
45   * </p>
46   *   
47   * @author The Apache Directory Project (dev@directory.apache.org)
48   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
49   */
50  public interface IoSession {

69       * operation is asynchronous; {@link IoHandler#messageSent(IoSession, Object)}
70       * will be invoked when the message is actually sent to remote peer.
71       * You can also ait for the returned {@link WriteFuture} if you want
72       * to wait for the session actually closed.
73       */
74      WriteFuture write( Object message );

76      /***
77       * Closes this session immediately.  This operation is asynthronous.
78       * Wait for the returned {@link CloseFuture} if you want to wait for
79       * the session actually closed.

81      CloseFuture close();

83      /***

85       * This method is identical with <tt>getAttribute( "" )</tt>.

87      Object getAttachment();

89      /***

91       * This method is identical with <tt>setAttribute( "", attachment )</tt>.

93       * @return Old attachment.  <tt>null</tt> if it is new.
94       */
95      Object setAttachment( Object attachment );
96      
97      /***
98       * Returns the value of user-defined attribute of this session.
99       *

102      */
103     Object getAttribute( String key );

108      * @param key the key of the attribute

110      * @return The old value of the attribute.  <tt>null</tt> if it is new.
111      */
112     Object setAttribute( String key, Object value );
113     
114     /***

275     long getLastReadTime();

285      */

289      * Returns the number of the fired continuous <tt>sessionIdle</tt> events

291      * <p>
292      * If <tt>sessionIdle</tt> event is fired first after some time after I/O,

304     long getLastIdleTime( IdleStatus status );
