39  /***
40   * An abstract implementation of {@link IoFilterChain} that provides
41   * common operations for developers to implement their own transport layer.
42   * <p>
43   * The only method a developer should implement is
44   * {@link #doWrite(IoSession, IoFilter.WriteRequest)}.  This method is invoked
45   * when filter chain is evaluated for
46   * {@link IoFilter#filterWrite(NextFilter, IoSession, IoFilter.WriteRequest)} and 
47   * finally to be written out.
48   * 
49   * @author The Apache Directory Project
50   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
51   */

61      protected AbstractIoFilterChain( IoSession session )
62      {
63          if( session == null )
64          {
65              throw new NullPointerException( "session" );
66          }
67          
68          this.session = session;
69          head = new EntryImpl( null, null, "head", createHeadFilter() );
70          tail = new EntryImpl( head, null, "tail", createTailFilter() );
71          head.nextEntry = tail;
72      }

120             public void filterWrite( NextFilter nextFilter, IoSession session,
121                                      WriteRequest writeRequest ) throws Exception
122             {
123                 if( session.getTransportType().getEnvelopeType().isAssignableFrom( writeRequest.getMessage().getClass() ) )
124                 {
125                     doWrite( session, writeRequest );
126                 }
127                 else
128                 {
129                     throw new IllegalStateException(
130                             "Write requests must be transformed to " + 
131                             session.getTransportType().getEnvelopeType() +
132                             ": " + writeRequest );
133                 }
134             }
135 
136             public void filterClose( NextFilter nextFilter, IoSession session, CloseFuture closeFuture ) throws Exception
137             {
138                 doClose( session, closeFuture );
139             }
140         };
141     }

308     private void register( EntryImpl prevEntry, String name, IoFilter filter )
309     {
310         EntryImpl newEntry = new EntryImpl( prevEntry, prevEntry.nextEntry, name, filter );
311 
312         IoFilterLifeCycleManager lifeCycleManager = IoFilterLifeCycleManager.getInstance();
313         
314         synchronized( lifeCycleManager )
315         {
316             lifeCycleManager.callInitIfNecessary( filter );
317 
318             try
319             {
320                 lifeCycleManager.callOnPreAdd( this, name, filter, newEntry.getNextFilter() );
321             }
322             finally
323             {
324                 lifeCycleManager.callDestroyIfNecessary( filter );
325             }
326 
327             prevEntry.nextEntry.prevEntry = newEntry;
328             prevEntry.nextEntry = newEntry;
329             name2entry.put( name, newEntry );
330 
331             try
332             {
333                 lifeCycleManager.callOnPostAdd( this, name, filter, newEntry.getNextFilter() );
334             }
335             catch( IoFilterLifeCycleException e )
336             {
337                 deregister0( newEntry );
338                 throw e;
339             }
340             finally
341             {
342                 lifeCycleManager.callDestroyIfNecessary( filter );
343             }
344         }
345     }
346     
347     private void deregister( EntryImpl entry )
348     {
349         IoFilter filter = entry.getFilter();
350         IoFilterLifeCycleManager lifeCycleManager = IoFilterLifeCycleManager.getInstance();
351         
352         lifeCycleManager.callOnPreRemove( this, entry.getName(), filter, entry.getNextFilter() );
353         
354         deregister0( entry );
355 
356         try
357         {
358             lifeCycleManager.callOnPostRemove(
359                     this, entry.getName(), filter, entry.getNextFilter() );
360         }
361         finally
362         {
363             lifeCycleManager.callDestroyIfNecessary( filter );
364         }
365     }
366 
367     private void deregister0( EntryImpl entry )
368     {
369         EntryImpl prevEntry = entry.prevEntry;
370         EntryImpl nextEntry = entry.nextEntry;
371         prevEntry.nextEntry = nextEntry;
372         nextEntry.prevEntry = prevEntry;
373 
374         name2entry.remove( entry.name );
375     }

642     public String toString()
643     {
644         StringBuffer buf = new StringBuffer();
645         buf.append( "{ " );
646         
647         boolean empty = true;
648         
649         EntryImpl e = head.nextEntry;
650         while( e != tail )
651         {
652             if( !empty )
653             {
654                 buf.append( ", " );
655             }
656             else
657             {
658                 empty = false;
659             }
660             
661             buf.append( '(' );
662             buf.append( e.getName() );
663             buf.append( ':' );
664             buf.append( e.getFilter() );
665             buf.append( ')' );
666 
667             e = e.nextEntry;
668         }
669 
670         if( empty )
671         {
672             buf.append( "empty" );
673         }
674         
675         buf.append( " }" );
676         
677         return buf.toString();
678     }

697     private class EntryImpl implements Entry

720             
721             this.prevEntry = prevEntry;
722             this.nextEntry = nextEntry;
723             this.name = name;
724             this.filter = filter;
725             this.nextFilter = new NextFilter()
726             {

728                 {
729                     Entry nextEntry = EntryImpl.this.nextEntry;
730                     callNextSessionCreated( nextEntry, session );
