39  /***
40   * A Thread-pooling filter.  This filter forwards {@link IoHandler} events
41   * to its thread pool.
42   * <p>
43   * This is an implementation of
44   * <a href="http://deuce.doc.wustl.edu/doc/pspdfs/lf.pdf">Leader/Followers
45   * thread pool</a> by Douglas C. Schmidt et al.
46   * 
47   * @author The Apache Directory Project (dev@directory.apache.org)
48   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
49   */
50  public class ThreadPoolFilter extends IoFilterAdapter
51  {
52      /***
53       * Default maximum size of thread pool (2G).
54       */
55      public static final int DEFAULT_MAXIMUM_POOL_SIZE = Integer.MAX_VALUE;
56  
57      /***
58       * Default keep-alive time of thread pool (1 min).
59       */
60      public static final int DEFAULT_KEEP_ALIVE_TIME = 60 * 1000;
61  
62      /***
63       * A queue which contains {@link Integer}s which represents reusable
64       * thread IDs.  {@link Worker} first checks this queue and then
65       * uses {@link #threadId} when no reusable thread ID is available.
66       */
67      private static final Queue threadIdReuseQueue = new Queue();
68      private static int threadId = 0;
69      
70      private static int acquireThreadId()
71      {
72          synchronized( threadIdReuseQueue )
73          {
74              Integer id = ( Integer ) threadIdReuseQueue.pop();
75              if( id == null )
76              {
77                  return ++ threadId;
78              }
79              else
80              {
81                  return id.intValue();
82              }
83          }
84      }
85      
86      private static void releaseThreadId( int id )
87      {
88          synchronized( threadIdReuseQueue )
89          {
90              threadIdReuseQueue.push( new Integer( id ) );
91          }
92      }

181     public void destroy()
182     {
183         shuttingDown = true;
184         int expectedPoolSize = 0;
185         while( getPoolSize() != expectedPoolSize )
186         {
187             List allWorkers;
188             synchronized( poolSizeLock )
189             {
190                 allWorkers = new ArrayList( this.allWorkers );
191             }
192             
193             // You may not interrupt the current thread.
194             if( allWorkers.remove( Thread.currentThread() ) )
195             {
196                 expectedPoolSize = 1;
197             }
198             
199             for( Iterator i = allWorkers.iterator(); i.hasNext(); )
200             {
201                 Worker worker = ( Worker ) i.next();
202                 while( worker.isAlive() )
203                 {
204                     worker.interrupt();
205                     try
206                     {
207                         // This timeout will help us from 
208                         // infinite lock-up and interrupt workers again.
209                         worker.join( 100 );
210                     }
211                     catch( InterruptedException e )
212                     {
213                     }
214                 }
215             }
216         }
217         
218         this.allSessionBuffers.clear();
219         this.unfetchedSessionBuffers.clear();
220         this.buffers.clear();
221         this.followers.clear();
222         this.leader = null;
223     }

243     private void fireEvent( NextFilter nextFilter, IoSession session,
244                               EventType type, Object data )
245     {
246         final BlockingQueue unfetchedSessionBuffers = this.unfetchedSessionBuffers;
247         final Set allSessionBuffers = this.allSessionBuffers;
248         final Event event = new Event( type, nextFilter, data );
249 
250         synchronized( unfetchedSessionBuffers )
251         {
252             final SessionBuffer buf = getSessionBuffer( session );
253             final Queue eventQueue = buf.eventQueue;
254 
255             synchronized( buf )
256             {
257                 eventQueue.push( event );
258             }
259 
260             if( !allSessionBuffers.contains( buf ) )
261             {
262                 allSessionBuffers.add( buf );
263                 unfetchedSessionBuffers.push( buf );
264             }
265         }
266     }

268     /***
269      * Implement this method to fetch (or pop) a {@link SessionBuffer} from
270      * the given <tt>unfetchedSessionBuffers</tt>.  The default implementation
271      * simply pops the buffer from it.  You could prioritize the fetch order.
272      * 
273      * @return A non-null {@link SessionBuffer}
274      */
275     protected SessionBuffer fetchSessionBuffer( Queue unfetchedSessionBuffers )
276     {
277         return ( SessionBuffer ) unfetchedSessionBuffers.pop();
278     }

280     private SessionBuffer getSessionBuffer( IoSession session )
281     {
282         final Map buffers = this.buffers;
283         SessionBuffer buf = ( SessionBuffer ) buffers.get( session );
284         if( buf == null )
285         {
286             synchronized( buffers )
287             {
288                 buf = ( SessionBuffer ) buffers.get( session );
289                 if( buf == null )
290                 {
291                     buf = new SessionBuffer( session );
292                     buffers.put( session, buf );
293                 }
294             }
295         }
296         return buf;
297     }
298 
299     private void removeSessionBuffer( SessionBuffer buf )
300     {
301         final Map buffers = this.buffers;
302         final IoSession session = buf.session;
303         synchronized( buffers )
304         {
305             buffers.remove( session );
306         }
307     }

331     private class Worker extends Thread
332     {
333         private final int id;
334         private final Object promotionLock = new Object();
335         private boolean dead;
336 
337         private Worker()
338         {
339             int id = acquireThreadId();
340             this.id = id;
341             this.setName( threadNamePrefix + '-' + id );
342             increasePoolSize( this );
343         }
344 
345         public boolean lead()
346         {
347             final Object promotionLock = this.promotionLock;
348             synchronized( promotionLock )
349             {
350                 if( dead )
351                 {
352                     return false;
353                 }
354 
355                 leader = this;
356                 promotionLock.notify();
357             }
358             
359             return true;
360         }
361 
362         public void run()
363         {
364             for( ;; )
365             {
366                 if( !waitForPromotion() )
367                     break;
368 
369                 SessionBuffer buf = fetchBuffer();
370                 giveUpLead();
371                 if( buf == null )
372                 {
373                     break;
374                 }
375 
376                 processEvents( buf );
377                 follow();
378                 releaseBuffer( buf );
379             }
380 
381             decreasePoolSize( this );
382             releaseThreadId( id );
383         }
384 
385         private SessionBuffer fetchBuffer()
386         {
387             BlockingQueue unfetchedSessionBuffers = ThreadPoolFilter.this.unfetchedSessionBuffers;
388             synchronized( unfetchedSessionBuffers )
389             {
390                 while( !shuttingDown )
391                 {
392                     try
393                     {
394                         unfetchedSessionBuffers.waitForNewItem();
395                     }
396                     catch( InterruptedException e )
397                     {
398                         continue;
399                     }
400 
401                     return ThreadPoolFilter.this.fetchSessionBuffer( unfetchedSessionBuffers );
402                 }
403             }
404 
405             return null;
406         }
407 
408         private void processEvents( SessionBuffer buf )
409         {
410             final IoSession session = buf.session;
411             final Queue eventQueue = buf.eventQueue;
412             for( ;; )
413             {
414                 Event event;
415                 synchronized( buf )
416                 {
417                     event = ( Event ) eventQueue.pop();
418                     if( event == null )
419                         break;
420                 }
421                 processEvent( event.getNextFilter(), session,
422                               event.getType(), event.getData() );
423             }
424         }
425 
426         private void follow()
427         {
428             final Object promotionLock = this.promotionLock;
429             final Stack followers = ThreadPoolFilter.this.followers;
430             synchronized( promotionLock )
431             {
432                 if( this != leader )
433                 {
434                     synchronized( followers )
435                     {
436                         followers.push( this );
437                     }
438                 }
439             }
440         }
441 
442         private void releaseBuffer( SessionBuffer buf )
443         {
444             final BlockingQueue unfetchedSessionBuffers = ThreadPoolFilter.this.unfetchedSessionBuffers;
445             final Set allSessionBuffers = ThreadPoolFilter.this.allSessionBuffers;
446             final Queue eventQueue = buf.eventQueue;
447 
448             synchronized( unfetchedSessionBuffers )
449             {
450                 if( eventQueue.isEmpty() )
451                 {
452                     allSessionBuffers.remove( buf );
453                     removeSessionBuffer( buf );
454                 }
455                 else
456                 {
457                     unfetchedSessionBuffers.push( buf );
458                 }
459             }
460         }
461 
462         private boolean waitForPromotion()
463         {
464             final Object promotionLock = this.promotionLock;
465 
466             long startTime = System.currentTimeMillis();
467             long currentTime = System.currentTimeMillis();
468             
469             synchronized( promotionLock )
470             {
471                 while( this != leader && !shuttingDown )
472                 {
473                     // Calculate remaining keep-alive time
474                     int keepAliveTime = getKeepAliveTime();
475                     if( keepAliveTime > 0 )
476                     {
477                         keepAliveTime -= ( currentTime - startTime );
478                     }
479                     else
480                     {
481                         keepAliveTime = Integer.MAX_VALUE;
482                     }
483                     
484                     // Break the loop if there's no remaining keep-alive time.
485                     if( keepAliveTime <= 0 )
486                     {
487                         break;
488                     }
489 
490                     // Wait for promotion
491                     try
492                     {
493                         promotionLock.wait( keepAliveTime );
494                     }
495                     catch( InterruptedException e )
496                     {
497                     }
498 
499                     // Update currentTime for the next iteration
500                     currentTime = System.currentTimeMillis();
501                 }
502 
503                 boolean timeToLead = this == leader && !shuttingDown;
504 
505                 if( !timeToLead )
506                 {
507                     // time to die
508                     synchronized( followers )
509                     {
510                         followers.remove( this );
511                     }
512 
513                     // Mark as dead explicitly when we've got promotionLock.
514                     dead = true;
515                 }
516 
517                 return timeToLead;
518             }
519         }
520 
521         private void giveUpLead()
522         {
523             final Stack followers = ThreadPoolFilter.this.followers;
524             Worker worker;
525             do
526             {
527                 synchronized( followers )
528                 {
529                     worker = ( Worker ) followers.pop();
530                 }
531 
532                 if( worker == null )
533                 {
534                     // Increase the number of threads if we
535                     // are not shutting down and we can increase the number.
536                     if( !shuttingDown
537                         && getPoolSize() < getMaximumPoolSize() )
538                     {
539                         worker = new Worker();
540                         worker.lead();
541                         worker.start();
542                     }
543 
544                     // This loop should end because:
545                     // 1) lead() is called already,
546                     // 2) or it is shutting down and there's no more threads left.
547                     break;
548                 }
549             }
550             while( !worker.lead() );
551         }

659     protected void processEvent( NextFilter nextFilter,
660                                  IoSession session, EventType type,
661                                  Object data )
662     {
663         if( type == EventType.RECEIVED )
664         {
665             nextFilter.messageReceived( session, data );
666             ByteBufferUtil.releaseIfPossible( data );
667         }
668         else if( type == EventType.SENT )
669         {
670             nextFilter.messageSent( session, data );
671             ByteBufferUtil.releaseIfPossible( data );
672         }
673         else if( type == EventType.EXCEPTION )
674         {
675             nextFilter.exceptionCaught( session, ( Throwable ) data );
676         }
677         else if( type == EventType.IDLE )
678         {
679             nextFilter.sessionIdle( session, ( IdleStatus ) data );
680         }
681         else if( type == EventType.OPENED )
682         {
683             nextFilter.sessionOpened( session );
684         }
685         else if( type == EventType.CLOSED )
686         {
687             nextFilter.sessionClosed( session );
688         }
689     }
