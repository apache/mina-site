38  /***
39   * An SSL filter that encrypts and decrypts the data exchanged in the session.
40   * This filter uses an {@link SSLEngine} which was introduced in Java 5, so 
41   * Java version 5 or above is mandatory to use this filter. And please note that
42   * this filter only works for TCP/IP connections.
43   * <p>
44   * This filter logs debug information using {@link SessionLog}.
45   * 
46   * <h2>Implementing StartTLS</h2>
47   * <p>
48   * You can use {@link #DISABLE_ENCRYPTION_ONCE} attribute to implement StartTLS:
49   * <pre>
50   * public void messageReceived(IoSession session, Object message) {
51   *    if (message instanceof MyStartTLSRequest) {
52   *        // Insert SSLFilter to get ready for handshaking
53   *        session.getFilterChain().addFirst(sslFilter);
54   *
55   *        // Disable encryption temporarilly.
56   *        // This attribute will be removed by SSLFilter
57   *        // inside the Session.write() call below.
58   *        session.setAttribute(SSLFilter.DISABLE_ENCRYPTION_ONCE, Boolean.TRUE);
59   *
60   *        // Write StartTLSResponse which won't be encrypted.
61   *        session.write(new MyStartTLSResponse(OK));
62   *        
63   *        // Now DISABLE_ENCRYPTION_ONCE attribute is cleared.
64   *        assert session.getAttribute(SSLFilter.DISABLE_ENCRYPTION_ONCE) == null;
65   *    }
66   * }
67   * </pre>
68   * 
69   * @author The Apache Directory Project (dev@directory.apache.org)
70   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
71   */
72  public class SSLFilter extends IoFilterAdapter

81       * A session attribute key that makes next one write request bypass
82       * this filter (not encrypting the data).  This is a marker attribute,

151     /***
152      * (Re)starts SSL session for the specified <tt>session</tt> if not started yet.
153      * Please note that SSL session is automatically started by default, and therefore
154      * you don't need to call this method unless you've used TLS closure.
155      * 
156      * @return <tt>true</tt> if the SSL session has been started, <tt>false</tt> if already started.
157      * @throws SSLException if failed to start the SSL session
158      */
159     public boolean startSSL( IoSession session ) throws SSLException
160     {
161         SSLHandler handler = getSSLSessionHandler( session );
162         synchronized( handler )
163         {
164             if( handler.isOutboundDone() )
165             {
166                 NextFilter nextFilter = ( NextFilter ) session.getAttribute( NEXT_FILTER );
167                 handler.destroy();
168                 handler.init();
169                 handler.handshake( nextFilter );
170                 return true;
171             }
172             else
173             {
174                 return false;
175             }
176         }
177     }

179     /***
180      * Returns <tt>true</tt> if and only if the specified <tt>session</tt> is
181      * encrypted/decrypted over SSL/TLS currently.  This method will start
182      * to retun <tt>false</tt> after TLS <tt>close_notify</tt> message
183      * is sent and any messages written after then is not goinf to get encrypted.
184      */
185     public boolean isSSLStarted( IoSession session )
186     {
187         SSLHandler handler = getSSLSessionHandler( session );
188         synchronized( handler )
189         {
190             return !handler.isOutboundDone();
191         }

195      * Stops the SSL session by sending TLS <tt>close_notify</tt> message to
196      * initiate TLS closure.
197      * 
198      * @param session the {@link IoSession} to initiate TLS closure
199      * @throws SSLException if failed to initiate TLS closure
200      * @throws IllegalArgumentException if this filter is not managing the specified session
201      */
202     public WriteFuture stopSSL( IoSession session ) throws SSLException
203     {
204         SSLHandler handler = getSSLSessionHandler( session );
205         NextFilter nextFilter = ( NextFilter ) session.getAttribute( NEXT_FILTER );
206         synchronized( handler )
207         {
208             return initiateClosure( nextFilter, session );
209         }

318     public void onPostAdd( IoFilterChain parent, String name, NextFilter nextFilter ) throws SSLException
319     {
320         IoSession session = parent.getSession();
321         session.setAttribute( NEXT_FILTER, nextFilter );
322         
323         // Create an SSL handler and start handshake.
324         SSLHandler handler =
325             new SSLHandler( this, sslContext, session );
326         session.setAttribute( SSL_HANDLER, handler );
327         handler.handshake( nextFilter );
328     }

330     public void onPreRemove( IoFilterChain parent, String name, NextFilter nextFilter ) throws SSLException
331     {
332         IoSession session = parent.getSession();
333         WriteFuture future = stopSSL( session );
334         session.removeAttribute( NEXT_FILTER );
335         session.removeAttribute( SSL_HANDLER );
336         future.join();
337     }

340     public void sessionClosed( NextFilter nextFilter, IoSession session ) throws SSLException
341     {
342         SSLHandler handler = getSSLSessionHandler( session );
343         try
344         {
345             synchronized( handler )
346             {
347                 if( isSSLStarted( session ) )
348                 {
349                     if( SessionLog.isDebugEnabled( session ) )
350                     {
351                         SessionLog.debug( session, " Closed: " + getSSLSessionHandler( session ) );
352                     }
353                 }
354                 
355                 // release resources
356                 handler.destroy();
357             }
358         }
359         finally
360         {
361            // notify closed session
362            nextFilter.sessionClosed( session );
363         }
364     }

366     public void messageReceived( NextFilter nextFilter, IoSession session,
367                                  Object message ) throws SSLException
368     {
369         SSLHandler handler = getSSLSessionHandler( session );
370         synchronized( handler )
371         {
372             if( !isSSLStarted( session ) )
373             {
374                 if( handler.isInboundDone() )
375                 {
376                     nextFilter.messageReceived( session, message );
377                     return;
378                 }
379             }
380     
381             ByteBuffer buf = ( ByteBuffer ) message;
382             if( SessionLog.isDebugEnabled( session ) )
383             {
384                 SessionLog.debug( session, " Data Read: " + handler + " (" + buf+ ')' );
385             }
386 
387             try
388             {
389                 // forward read encrypted data to SSL handler
390                 handler.messageReceived( nextFilter, buf.buf() );
391 
392                 // Handle data to be forwarded to application or written to net
393                 handleSSLData( nextFilter, handler );
394 
395                 if( handler.isInboundDone() )
396                 {
397                     if( handler.isOutboundDone() )
398                     {
399                         if( SessionLog.isDebugEnabled( session ) )
400                         {
401                             SessionLog.debug(
402                                     session, " SSL Session closed." );
403                         }
404                         
405                         handler.destroy();
406                     }
407                     else
408                     {
409                         initiateClosure( nextFilter, session );
410                     }
411 
412                     if( buf.hasRemaining() )
413                     {
414                         nextFilter.messageReceived( session, buf );
415                     }
416                 }
417             }
418             catch( SSLException ssle )
419             {
420                 if( !handler.isInitialHandshakeComplete() )
421                 {
422                     SSLException newSSLE = new SSLHandshakeException(
423                             "Initial SSL handshake failed." );
424                     newSSLE.initCause( ssle );
425                     ssle = newSSLE;
426                 }
427 
428                 throw ssle;
429             }
430         }
431     }

448     public void filterWrite( NextFilter nextFilter, IoSession session, WriteRequest writeRequest ) throws SSLException
449     {
450         SSLHandler handler = getSSLSessionHandler( session );
451         synchronized( handler )
452         {
453             if( !isSSLStarted( session ) )
454             {
455                 nextFilter.filterWrite( session, writeRequest );
456                 return;
457             }
458             
459             // Don't encrypt the data if encryption is disabled.
460             if( session.containsAttribute( DISABLE_ENCRYPTION_ONCE ) )
461             {
462                 // Remove the marker attribute because it is temporary.
463                 session.removeAttribute( DISABLE_ENCRYPTION_ONCE );
464                 nextFilter.filterWrite( session, writeRequest );
465                 return;
466             }
467             
468             // Otherwise, encrypt the buffer.
469             ByteBuffer buf = ( ByteBuffer ) writeRequest.getMessage();
470         
471             if( SessionLog.isDebugEnabled( session ) )
472             {
473                 SessionLog.debug( session, " Filtered Write: " + handler );
474             }
475 
476             if( handler.isWritingEncryptedData() )
477             {
478                 // data already encrypted; simply return buffer
479                 if( SessionLog.isDebugEnabled( session ) )
480                 {
481                     SessionLog.debug( session, "   already encrypted: " + buf );
482                 }
483                 nextFilter.filterWrite( session, writeRequest );
484                 return;
485             }
486             
487             if( handler.isInitialHandshakeComplete() )
488             {
489                 // SSL encrypt
490                 if( SessionLog.isDebugEnabled( session ) )
491                 {
492                     SessionLog.debug( session, " encrypt: " + buf );
493                 }
494                 
495                 int pos = buf.position();
496                 handler.encrypt( buf.buf() );
497                 buf.position( pos );
498                 ByteBuffer encryptedBuffer = new EncryptedBuffer(
499                         SSLHandler.copy( handler.getOutNetBuffer() ), buf );
500 
501                 if( SessionLog.isDebugEnabled( session ) )
502                 {
503                     SessionLog.debug( session, " encrypted buf: " + encryptedBuffer);
504                 }
505                 nextFilter.filterWrite( session, new WriteRequest( encryptedBuffer, writeRequest.getFuture() ) );
506                 return;
507             }
508             else
509             {
510                 if( !session.isConnected() )
511                 {
512                     if( SessionLog.isDebugEnabled( session ) )
513                     {
514                         SessionLog.debug( session, " Write request on closed session." );
515                     }
516                 }
517                 else
518                 {
519                     if( SessionLog.isDebugEnabled( session ) )
520                     {
521                         SessionLog.debug( session, " Handshaking is not complete yet. Buffering write request." );
522                     }
523                     handler.scheduleWrite( nextFilter, writeRequest );
524                 }
525             }
526         }
527     }

529     public void filterClose( NextFilter nextFilter, IoSession session, CloseFuture closeFuture ) throws SSLException
530     {
531         SSLHandler handler = getSSLSessionHandler( session );
532 
533         WriteFuture future = null;
534         synchronized( handler )
535         {
536             try
537             {
538                 if( isSSLStarted( session ) )
539                 {
540                     future = initiateClosure( nextFilter, session );
541                 }
542             }
543             finally
544             {
545                 nextFilter.filterClose( session, closeFuture );
546             }
547         }
548         
549         if( future != null )
550         {
551             future.join();
552         }
553     }

555     private WriteFuture initiateClosure( NextFilter nextFilter, IoSession session ) throws SSLException
556     {
557         SSLHandler handler = getSSLSessionHandler( session );
558         // if already shut down
559         if( !handler.closeOutbound() )
560         {
561             return WriteFuture.newNotWrittenFuture();
562         }
563         
564         // there might be data to write out here?
565         WriteFuture future = handler.writeNetBuffer( nextFilter );
566         
567         if( handler.isInboundDone() )
568         {
569             handler.destroy();
570         }
571 
572         if( session.containsAttribute( USE_NOTIFICATION ) )
573         {
574             nextFilter.messageReceived( session, SESSION_UNSECURED );
575         }
576         
577         return future;
578     }

582     private void handleSSLData( NextFilter nextFilter, SSLHandler handler ) throws SSLException
583     {
584         // Flush any buffered write requests occurred before handshaking.
585         if( handler.isInitialHandshakeComplete() )
586         {
587             handler.flushScheduledWrites();
588         }
589 
590         // Write encrypted data to be written (if any)
591         handler.writeNetBuffer( nextFilter );
592 
593         // handle app. data read (if any)
594         handleAppDataRead( nextFilter, handler );
595     }
596 
597     private void handleAppDataRead( NextFilter nextFilter, SSLHandler handler )
598     {
599         IoSession session = handler.getSession();
600         if( !handler.getAppBuffer().hasRemaining() )
601         {
602             return;
603         }
604 
605         if( SessionLog.isDebugEnabled( session ) )
606         {
607             SessionLog.debug( session, " appBuffer: " + handler.getAppBuffer() );
608         }
609 
610         // forward read app data
611         ByteBuffer readBuffer = SSLHandler.copy( handler.getAppBuffer() );
612         if( SessionLog.isDebugEnabled( session ) )
613         {
614             SessionLog.debug( session, " app data read: " + readBuffer + " (" + readBuffer.getHexDump() + ')' );
615         }
616         nextFilter.messageReceived( session, readBuffer );
617     }

619     private SSLHandler getSSLSessionHandler( IoSession session )
620     {
621         SSLHandler handler = ( SSLHandler ) session.getAttribute( SSL_HANDLER );
622         if( handler == null )
623         {
624             throw new IllegalStateException();
625         }
626         if( handler.getParent() != this )
627         {
628             throw new IllegalArgumentException( "Not managed by this filter." );
629         }
630         return handler;
631     }
