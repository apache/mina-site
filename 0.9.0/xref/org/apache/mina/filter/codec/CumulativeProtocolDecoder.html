19  package org.apache.mina.filter.codec;
20  
21  import org.apache.mina.common.ByteBuffer;
22  import org.apache.mina.common.IoSession;
23  
24  /***
25   * A {@link ProtocolDecoder} that cumulates the content of received
26   * buffers to a <em>cumulative buffer</em> to help users implement decoders.
27   * <p>
28   * If the received {@link ByteBuffer} is only a part of a message.
29   * decoders should cumulate received buffers to make a message complete or
30   * to postpone decoding until more buffers arrive.
31   * <p>
32   * Here is an example decoder that decodes a list of integers:
33   * <pre>
34   * public class IntegerDecoder extends CumulativeProtocolDecoder {
35   * 
36   *     public IntegerDecoder() {
37   *         super(4);
38   *     }
39   * 
40   *     protected boolean doDecode(IoSession session, ByteBuffer in,
41   *                                ProtocolDecoderOutput out) throws ProtocolViolationException {
42   *         if (in.remaining() < 4) {
43   *             return false; // Cumulate remainder to decode later.
44   *         }
45   *         
46   *         out.write(new Integer(in.getInt()));
47   * 
48   *         // Decoded one integer; CumulativeProtocolDecoder will call me again,
49   *         // so I can decode as many integers as possible.
50   *         return true;
51   *     }
52   * }
53   * </pre>
54   * 
55   * @author The Apache Directory Project (dev@directory.apache.org)
56   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
57   */
58  public abstract class CumulativeProtocolDecoder extends ProtocolDecoderAdapter {
59      
60      private static final String BUFFER = CumulativeProtocolDecoder.class.getName() + ".Buffer";
61      
62      private final int initialCapacity;
63      
64      /***
65       * Creates a new instance with the 16 bytes initial capacity of
66       * cumulative buffer.  Please note that the capacity increases
67       * automatically.
68       */
69      protected CumulativeProtocolDecoder()
70      {
71          this( 16 );
72      }
73      
74      /***
75       * Creates a new instance with the specified initial capacity of
76       * cumulative buffer.  Please note that the capacity increases
77       * automatically.
78       */
79      protected CumulativeProtocolDecoder( int initialCapacity )
80      {
81          if( initialCapacity < 0 )
82          {
83              throw new IllegalArgumentException( "initialCapacity: " + initialCapacity );
84          }
85          
86          this.initialCapacity = initialCapacity;
87      }
88      
89      /***
90       * Cumulates content of <tt>in</tt> into internal buffer and forwards
91       * decoding request to {@link #doDecode(IoSession, ByteBuffer, ProtocolDecoderOutput)}.
92       * <tt>doDecode()</tt> is invoked repeatedly until it returns <tt>false</tt>
93       * and the cumulative buffer is compacted after decoding ends.
94       * 
95       * @throws IllegalStateException if your <tt>doDecode()</tt> returned
96       *                               <tt>true</tt> not consuming the cumulative buffer.
97       */
98      public void decode( IoSession session, ByteBuffer in,
99                          ProtocolDecoderOutput out ) throws Exception
100     {
101         ByteBuffer buf = ( ByteBuffer ) session.getAttribute( BUFFER );
102         if( buf == null )
103         {
104             buf = ByteBuffer.allocate( initialCapacity );
105             buf.setAutoExpand( true );
106             session.setAttribute( BUFFER, buf );
107         }
108         
109         buf.put( in );
110         buf.flip();
111 
112         try
113         {
114             for( ;; )
115             {
116                 int oldPos = buf.position();
117                 boolean decoded = doDecode( session, buf, out );
118                 if( decoded )
119                 {
120                     if( buf.position() == oldPos )
121                     {
122                         throw new IllegalStateException(
123                                 "doDecode() can't return true when buffer is not consumed." );
124                     }
125                     
126                     if( !buf.hasRemaining() )
127                     {
128                         break;
129                     }
130                 }
131                 else
132                 {
133                     break;
134                 }
135             }
136         }
137         finally
138         {
139             buf.compact();
140         }
141     }
142     
143     /***
144      * Implement this method to consume the specified cumulative buffer and
145      * decode its content into message(s). 
146      *  
147      * @param in the cumulative buffer
148      * @return <tt>true</tt> if and only if there's more to decode in the buffer
149      *         and you want to have <tt>doDecode</tt> method invoked again.
150      *         Return <tt>false</tt> if remaining data is not enough to decode,
151      *         then this method will be invoked again when more data is cumulated.
152      * @throws Exception if cannot decode <tt>in</tt>.
153      */
154     protected abstract boolean doDecode( IoSession session, ByteBuffer in,
155                                          ProtocolDecoderOutput out ) throws Exception;
156 
157     /***
158      * Releases the cumulative buffer used by the specified <tt>session</tt>.
159      * Please don't forget to call <tt>super.dispose( session )</tt> when
160      * you override this method.
161      */
162     public void dispose( IoSession session ) throws Exception
163     {
164         ByteBuffer buf = ( ByteBuffer ) session.getAttribute( BUFFER );
165         if( buf != null )
166         {
167             buf.release();
168             session.removeAttribute( BUFFER );
169         }
170     }
171 }
