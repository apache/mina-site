19  package org.apache.mina.filter.codec;
20  
21  import org.apache.mina.common.ByteBuffer;
22  
23  /***
24   * An exception that is thrown when {@link ProtocolDecoder}
25   * cannot understand or failed to validate the specified {@link ByteBuffer}
26   * content.
27   * 
28   * @author The Apache Directory Project (dev@directory.apache.org)
29   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
30   */
31  public class ProtocolDecoderException extends ProtocolCodecException
32  {
33      private static final long serialVersionUID = 3545799879533408565L;
34  
35  	private String hexdump;
36  
37      /***
38       * Constructs a new instance.
39       */
40      public ProtocolDecoderException()
41      {
42      }
43  
44      /***
45       * Constructs a new instance with the specified message.
46       */
47      public ProtocolDecoderException( String message )
48      {
49          super( message );
50      }
51  
52      /***
53       * Constructs a new instance with the specified cause.
54       */
55      public ProtocolDecoderException( Throwable cause )
56      {
57          super( cause );
58      }
59  
60      /***
61       * Constructs a new instance with the specified message and the specified
62       * cause.
63       */
64      public ProtocolDecoderException( String message, Throwable cause )
65      {
66          super( message, cause );
67      }
68  
69      /***
70       * Returns the message and the hexdump of the unknown part.
71       */
72      public String getMessage()
73      {
74          String message = super.getMessage();
75  
76          if( message == null )
77          {
78              message = "";
79          }
80  
81          if( hexdump != null )
82          {
83              return message + ( ( message.length() > 0 ) ? " " : "" )
84                     + "(Hexdump: " + hexdump + ')';
85          }
86          else
87          {
88              return message;
89          }
90      }
91  
92      /***
93       * Returns the hexdump of the unknown message part.
94       */
95      public String getHexdump()
96      {
97          return hexdump;
98      }
99  
100     /***
101      * Sets the hexdump of the unknown message part.
102      */
103     public void setHexdump( String hexdump )
104     {
105         if( this.hexdump != null )
106         {
107             throw new IllegalStateException( "Hexdump cannot be set more than once." );
108         }
109         this.hexdump = hexdump;
110     }
111 }
