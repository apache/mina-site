24  /***
25   * A simple implementation of {@link ServiceRegistry}.
26   * 
27   * This service registry supports socket, datagram, VM-pipe transport types,
28   * and thread pools were added by default. 
29   * 
30   * @author The Apache Directory Project (dev@directory.apache.org)
31   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $, 
32   */
33  public class SimpleServiceRegistry implements ServiceRegistry
34  {
35      protected final IoAcceptor socketIoAcceptor = new SocketAcceptor();
36      protected final IoAcceptor datagramIoAcceptor = new DatagramAcceptor();
37      protected final IoAcceptor vmPipeAcceptor = new VmPipeAcceptor();
38      protected final ThreadPoolFilter threadPoolFilter = new ThreadPoolFilter();
39      private final Set services = new HashSet();
40  
41      public SimpleServiceRegistry()
42      {
43      }
44  
45      public void bind( Service service, IoHandler handler ) throws IOException
46      {
47          bind( service, handler, null );
48      }
49      
50      public void bind( Service service, IoHandler handler, IoFilterChainBuilder filterChainBuilder ) throws IOException
51      {
52          IoAcceptor acceptor = findAcceptor( service.getTransportType() );
53          if( filterChainBuilder == null )
54          {
55              filterChainBuilder = IoFilterChainBuilder.NOOP;
56          }
57          acceptor.bind(
58                  service.getAddress(), handler,
59                  new IoFilterChainBuilderWrapper( service, filterChainBuilder ) );
60          services.add( service );
61      }
62  
63      public synchronized void unbind( Service service )
64      {
65          IoAcceptor acceptor = findAcceptor( service
66                  .getTransportType() );
67          try
68          {
69              acceptor.unbind( service.getAddress() );
70          }
71          catch( Exception e )
72          {
73              // ignore
74          }
75          
76          services.remove( service );
77      }
78      
79      public synchronized void unbindAll()
80      {
81          Iterator it = new HashSet( services ).iterator();
82          while( it.hasNext() )
83          {
84              Service s = ( Service ) it.next();
85              unbind( s );
86          }
87      }
88  
89      public IoAcceptor getAcceptor( TransportType transportType )
90      {
91          return findAcceptor( transportType );
92      }
93  
94      public synchronized Set getAllServices()
95      {
96          return new HashSet( services );
97      }
98  
99      public synchronized Set getServices( String name )
100     {
101         Set result = new HashSet();
102         Iterator it = services.iterator();
103         while( it.hasNext() )
104         {
105             Service s = ( Service ) it.next();
106             if( name.equals( s.getName() ) )
107             {
108                 result.add( s );
109             }
110         }
111         return result;
112     }
113 
114     public Set getServices( TransportType transportType )
115     {
116         Set result = new HashSet();
117         Iterator it = services.iterator();
118         while( it.hasNext() )
119         {
120             Service s = ( Service ) it.next();
121             if( s.getTransportType() == transportType )
122             {
123                 result.add( s );
124             }
125         }
126         return result;
127     }
128 
129     public Set getServices( int port )
130     {
131         Set result = new HashSet();
132         Iterator it = services.iterator();
133         while( it.hasNext() )
134         {
135             Service s = ( Service ) it.next();
136             SocketAddress addr = s.getAddress();
137             int servicePort;
138             
139             if( addr instanceof InetSocketAddress )
140             {
141                 servicePort = ( ( InetSocketAddress ) addr ).getPort();
142             }
143             else if( addr instanceof VmPipeAddress )
144             {
145                 servicePort = ( ( VmPipeAddress ) addr ).getPort();
146             }
147             else
148             {
149                 servicePort = -1; // this cannot happen 
150             }
151             
152             if( servicePort == port )
153             {
154                 result.add( s );
155             }
156         }
157         return result;
158     }
159 
160     protected IoAcceptor findAcceptor( TransportType transportType )
161     {
162         if( transportType == TransportType.SOCKET )
163             return socketIoAcceptor;
164         else if( transportType == TransportType.DATAGRAM )
165             return datagramIoAcceptor;
166         else
167             return vmPipeAcceptor;
168 
169     }
170     
171     private class IoFilterChainBuilderWrapper implements IoFilterChainBuilder
172     {
173         private final Service service;
174         private final IoFilterChainBuilder originalBuilder;
175         
176         private IoFilterChainBuilderWrapper( Service service, IoFilterChainBuilder originalBuilder )
177         {
178             this.service = service;
179             this.originalBuilder = originalBuilder;
180         }
181         
182         public void buildFilterChain( IoFilterChain chain ) throws Exception
183         {
184             chain.getSession().setAttribute( SERVICE, service );
185 
186             try
187             {
188                 originalBuilder.buildFilterChain( chain );
189             }
190             finally
191             {
192                 chain.addFirst( "threadPool", threadPoolFilter );
193             }
194         }
195     }
