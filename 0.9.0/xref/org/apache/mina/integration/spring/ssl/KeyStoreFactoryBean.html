19  package org.apache.mina.integration.spring.ssl;
20  
21  import java.io.BufferedInputStream;
22  import java.io.File;
23  import java.io.FileInputStream;
24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.security.KeyStore;
27  
28  import org.springframework.beans.factory.config.AbstractFactoryBean;
29  import org.springframework.core.io.Resource;
30  import org.springframework.util.Assert;
31  
32  /***
33   * Spring {@link org.springframework.beans.factory.FactoryBean} implementation 
34   * which makes it possible to configure {@link java.security.KeyStore} instances
35   * using Spring.
36   *
37   * @author The Apache Directory Project (dev@directory.apache.org)
38   * @version $Rev$, $Date$
39   */
40  public class KeyStoreFactoryBean extends AbstractFactoryBean
41  {
42      private String type = "JKS";
43      private String provider = null;
44      private char[] password = null;
45      private File file = null;
46      private Resource resource = null;
47  
48      /***
49       * Creates a new {@link KeyStore}. This method will be called
50       * by the base class when Spring creates a bean using this FactoryBean.
51       * 
52       * @return the {@link KeyStore} instance.
53       */
54      protected Object createInstance() throws Exception
55      {
56          if( file == null && resource == null )
57          {
58              throw new IllegalArgumentException( "Required property missing. " +
59                      "Either 'file' or 'resource' have to be specified" );
60          }
61          
62          KeyStore ks = null;
63          if( provider == null )
64          {
65              ks = KeyStore.getInstance( type );
66          }
67          else
68          {
69              ks = KeyStore.getInstance( type, provider );
70          }
71          
72          InputStream is = null;
73          if( file != null )
74          {
75              is = new BufferedInputStream( new FileInputStream( file ) );
76          }
77          else
78          {
79              is = resource.getInputStream();
80          }
81          
82          try
83          {
84              ks.load( is, password );
85          }
86          finally
87          {
88              try
89              {
90                  is.close();
91              }
92              catch( IOException ignored )
93              {
94              }
95          }
96          
97          return ks;
98      }
99  
100     public Class getObjectType()
101     {
102         return KeyStore.class;
103     }
104 
105     /***
106      * Sets the file which contains the key store. Either this
107      * property or {@link #setProvider(String)} have to be set.
108      * 
109      * @param file the file to load the key store from.
110      */
111     public void setFile( File file )
112     {
113         this.file = file;
114     }
115 
116     /***
117      * Sets the key store password. If this value is <code>null</code> no
118      * password will be used to check the integrity of the key store.
119      * 
120      * @param password the password or <code>null</code> if no password is 
121      *        needed.
122      */
123     public void setPassword( String password )
124     {
125         if( password != null )
126         {
127             this.password = password.toCharArray();
128         }
129         else
130         {
131             this.password = null;
132         }
133     }
134 
135     /***
136      * Sets the name of the provider to use when creating the key store. The
137      * default is to use the platform default provider.
138      * 
139      * @param provider the name of the provider, e.g. SUN.
140      */
141     public void setProvider( String provider )
142     {
143         this.provider = provider;
144     }
145 
146     /***
147      * Sets a Spring {@link Resource} which contains the key store. Either this
148      * property or {@link #setFile(File)} have to be set.
149      * 
150      * @param resource the resource to load the key store from.
151      */
152     public void setResource( Resource resource )
153     {
154         this.resource = resource;
155     }
156 
157     /***
158      * Sets the type of key store to create. The default is to create a 
159      * JKS key store.
160      * 
161      * @param type the type to use when creating the key store.
162      * @throws IllegalArgumentException if the specified value is 
163      *         <code>null</code>.
164      */
165     public void setType( String type )
166     {
167         Assert.notNull( type, "Property 'type' may not be null" );
168         this.type = type;
169     }
