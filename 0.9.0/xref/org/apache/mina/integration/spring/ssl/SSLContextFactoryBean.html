33  /***
34   * Spring {@link org.springframework.beans.factory.FactoryBean} implementation 
35   * which makes it possible to configure {@link javax.net.ssl.SSLContext} 
36   * instances using Spring.
37   * <p>
38   * If no properties are set the returned {@link javax.net.ssl.SSLContext} will
39   * be equivalent to what the following creates:
40   * <pre>
41   *      SSLContext c = SSLContext.getInstance( "TLS" );
42   *      c.init( null, null, null );
43   * </pre>
44   * </p>
45   * <p>
46   * Use the properties prefixed with <code>keyManagerFactory</code> to control
47   * the creation of the {@link javax.net.ssl.KeyManager} to be used.
48   * </p>
49   * <p>
50   * Use the properties prefixed with <code>trustManagerFactory</code> to control
51   * the creation of the {@link javax.net.ssl.TrustManagerFactory} to be used.
52   * </p>
53   *
54   * @author The Apache Directory Project (dev@directory.apache.org)
55   * @version $Rev$, $Date$
56   */
57  public class SSLContextFactoryBean extends AbstractFactoryBean
58  {
59      private String protocol = "TLS";
60      private String provider = null;
61      private SecureRandom secureRandom = null;
62      
63      private KeyStore keyManagerFactoryKeyStore = null;
64      private char[] keyManagerFactoryKeyStorePassword = null;
65      private KeyManagerFactory keyManagerFactory = null;
66      private String keyManagerFactoryAlgorithm = null;
67      private String keyManagerFactoryProvider = null;
68      private boolean keyManagerFactoryAlgorithmUseDefault = false;
69  
70      private KeyStore trustManagerFactoryKeyStore = null;
71      private TrustManagerFactory trustManagerFactory = null;
72      private String trustManagerFactoryAlgorithm = null;
73      private String trustManagerFactoryProvider = null;
74      private boolean trustManagerFactoryAlgorithmUseDefault = false;
75      
76      protected Object createInstance() throws Exception
77      {
78          KeyManagerFactory kmf = this.keyManagerFactory;
79          TrustManagerFactory tmf = this.trustManagerFactory;
80          
81          if( kmf == null )
82          {
83              String algorithm = keyManagerFactoryAlgorithm;
84              if( algorithm == null && keyManagerFactoryAlgorithmUseDefault )
85              {
86                  algorithm = KeyManagerFactory.getDefaultAlgorithm();
87              }
88              if( algorithm != null )
89              {
90                  if( keyManagerFactoryProvider == null )
91                  {
92                      kmf = KeyManagerFactory.getInstance( algorithm );
93                  }
94                  else
95                  {
96                      kmf = KeyManagerFactory.getInstance( algorithm, 
97                                                      keyManagerFactoryProvider );
98                  }
99              }
100         }
101         
102         if( tmf == null )
103         {
104             String algorithm = trustManagerFactoryAlgorithm;
105             if( algorithm == null && trustManagerFactoryAlgorithmUseDefault )
106             {
107                 algorithm = TrustManagerFactory.getDefaultAlgorithm();
108             }
109             if( algorithm != null )
110             {
111                 if( trustManagerFactoryProvider == null )
112                 {
113                     tmf = TrustManagerFactory.getInstance( algorithm );
114                 }
115                 else
116                 {
117                     tmf = TrustManagerFactory.getInstance( algorithm, 
118                                                   trustManagerFactoryProvider );
119                 }
120             }
121         }
122         
123         KeyManager[] keyManagers = null; 
124         if( kmf != null )
125         {
126             kmf.init( keyManagerFactoryKeyStore, 
127                       keyManagerFactoryKeyStorePassword );
128             keyManagers = kmf.getKeyManagers();
129         }
130         TrustManager[] trustManagers = null; 
131         if( tmf != null )
132         {
133             tmf.init( trustManagerFactoryKeyStore );
134             trustManagers = tmf.getTrustManagers();
135         }
136         
137         SSLContext context = null;
138         if( provider == null )
139         {
140             context = SSLContext.getInstance( protocol );
141         }
142         else
143         {
144             context = SSLContext.getInstance( protocol, provider );
145         }
146         
147         context.init( keyManagers, trustManagers, secureRandom );
148         
149         return context;
150     }
