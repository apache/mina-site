19  package org.apache.mina.integration.spring.support;
20  
21  import java.net.SocketAddress;
22  
23  import org.apache.mina.common.DefaultIoFilterChainBuilder;
24  import org.apache.mina.common.IoAcceptor;
25  import org.apache.mina.common.IoFilterChainBuilder;
26  import org.apache.mina.common.IoHandler;
27  import org.apache.mina.common.IoSessionManager;
28  import org.apache.mina.integration.spring.Binding;
29  import org.apache.mina.integration.spring.IoFilterMapping;
30  import org.springframework.util.Assert;
31  
32  /***
33   * Abstract Spring FactoryBean which creates {@link IoAcceptor} instances and 
34   * enables their bindings and filters to be configured using Spring.
35   * <p>
36   * NOTE: Do NOT call {@link IoAcceptor#bind(SocketAddress, IoHandler)} on the
37   * created {@link IoAcceptor}. If you do add additional bindings that way
38   * {@link #destroyInstance(Object)} will not be able to unbind all bindings of
39   * the {@link IoAcceptor} and it will not be fully shut down when the Spring 
40   * BeanFactory is closed.
41   * </p>
42   * 
43   * @author The Apache Directory Project (dev@directory.apache.org)
44   * @version $Rev: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
45   */
46  public abstract class AbstractIoAcceptorFactoryBean extends
47          AbstractIoSessionManagerFactoryBean
48  {
49  
50      protected Binding[] bindings = new Binding[ 0 ];
51  
52      /***
53       * Creates the {@link IoAcceptor} configured by this factory bean.
54       * 
55       * @return the {@link IoAcceptor}.
56       * @throws Exception on errors.
57       */
58      protected abstract IoAcceptor createIoAcceptor() throws Exception;
59  
60      /***
61       * Creates a new {@link IoAcceptor}. Calls {@link #createIoAcceptor()} to 
62       * get the new {@link IoAcceptor} instance and then calls
63       * {@link AbstractIoSessionManagerFactoryBean#initIoSessionManager(IoSessionManager)}
64       * followed by {@link #initIoAcceptor(IoAcceptor)}.
65       * 
66       * @return the {@link IoAcceptor} instance.
67       */
68      protected Object createInstance() throws Exception
69      {
70          IoAcceptor acceptor = createIoAcceptor();
71  
72          initIoSessionManager( acceptor );
73          initIoAcceptor( acceptor );
74  
75          return acceptor;
76      }
77  
78      /***
79       * Initializes an {@link IoAcceptor} configured by this factory bean by
80       * calling {@link IoAcceptor#bind(SocketAddress, IoHandler, IoFilterChainBuilder)}
81       * for all bindings configured using {@link #setBindings(Binding[])}.
82       * 
83       * @param acceptor the {@link IoAcceptor}.
84       * @throws Exception on errors.
85       */
86      protected void initIoAcceptor( IoAcceptor acceptor ) throws Exception
87      {
88  
89          /*
90           * Bind all.
91           */
92          for( int i = 0; i < bindings.length; i++ )
93          {
94              Binding b = bindings[ i ];
95              DefaultIoFilterChainBuilder chainBuilder = 
96                  new DefaultIoFilterChainBuilder();
97              
98              IoFilterMapping[] fm = b.getFilterMappings();
99              for( int j = 0; j < fm.length; j++ )
100             {
101                 chainBuilder.addLast( fm[ j ].getName(), fm[ j ].getFilter() );
102             }
103             
104             SocketAddress address = parseSocketAddress( b.getAddress() );
105             acceptor.bind( address, b.getHandler(), chainBuilder );
106         }
107     }
108 
109     /***
110      * Destroys an {@link IoAcceptor} created by the factory bean by calling
111      * {@link #destroyIoAcceptor(IoAcceptor)} and then
112      * {@link AbstractIoSessionManagerFactoryBean#destroyIoSessionManager(IoSessionManager)}.
113      * This method may be overridden by extending classes if additional calls
114      * are necessary to shutdown the {@link IoAcceptor} or if the sequence of 
115      * calls should be different.
116      * <p>
117      * This method will be called by Spring when the BeanFactory creating this
118      * instance is closed. Spring will NOT call this method if this factory bean
119      * has been configured for non-singleton use.
120      * </p>
121      * 
122      * @param instance the {@link IoAcceptor} instance to be destroyed.
123      */
124     protected void destroyInstance( Object instance ) throws Exception
125     {
126         IoAcceptor acceptor = ( IoAcceptor ) instance;
127         destroyIoAcceptor( acceptor );
128         destroyIoSessionManager( acceptor );
129     }
130 
131     /***
132      * Destroys an {@link IoAcceptor} created by the factory bean by unbinding all
133      * bindings set through {@link #setBindings(Binding[])}.
134      * 
135      * @param acceptor the {@link IoAcceptor} instance to be destroyed.
136      */
137     protected void destroyIoAcceptor( IoAcceptor acceptor ) throws Exception
138     {
139         /*
140          * Unbind all.
141          */
142         for( int i = 0; i < bindings.length; i++ )
143         {
144             Binding b = bindings[ i ];
145             try
146             {
147                 acceptor.unbind( parseSocketAddress( b.getAddress() ) );
148             }
149             catch( Exception ignored )
150             {
151             }
152         }
153     }
154 
155     public Class getObjectType()
156     {
157         return IoAcceptor.class;
158     }
159 
160     /***
161      * Sets the bindings to be used by the {@link IoAcceptor} created by this 
162      * factory bean.
163      * 
164      * @param bindings the bindings.
165      * @throws IllegalArgumentException if the specified value is 
166      *         <code>null</code>.
167      * @see IoAcceptor#bind(SocketAddress, IoHandler, IoFilterChainBuilder)
168      * @see #parseSocketAddress(String)
169      * @see Binding
170      */
171     public void setBindings( Binding[] bindings )
172     {
173         Assert.notNull( bindings, "Property 'bindings' may not be null" );
174         this.bindings = bindings;
175     }
176 
177     /***
178      * Parses the specified string and returns the corresponding 
179      * {@link SocketAddress}.
180      * 
181      * @param s the string to parse.
182      * @return the {@link SocketAddress}.
183      */
184     protected abstract SocketAddress parseSocketAddress( String s );
185 }
