19  package org.apache.mina.integration.spring;
20  
21  import java.beans.PropertyEditorSupport;
22  import java.net.InetAddress;
23  import java.net.UnknownHostException;
24  
25  /***
26   * Java Bean PropertyEditor which converts Strings into InetAddress objects.
27   * This may be used together with Spring to be able to set InetAddress bean
28   * properties, e.g.
29   * {@link org.apache.mina.filter.BlacklistFilter#setBlacklist(InetAddress[])}.
30   * Simply calls {@link java.net.InetAddress#getByName(java.lang.String)} when
31   * converting from a String.
32   * <p>
33   * Use Spring's CustomEditorConfigurer to use this property editor in a Spring
34   * context. See chapter 3.14 of the Spring Reference Documentation for more
35   * info.
36   * </p>
37   * 
38   * @author The Apache Directory Project (dev@directory.apache.org)
39   * @version $Revision: 355016 $, $Date: 2005-12-08 16:00:30 +0900 (Thu, 08 Dec 2005) $
40   * 
41   * @see java.net.InetAddress
42   */
43  public class InetAddressEditor extends PropertyEditorSupport
44  {
45      public void setAsText( String text ) throws IllegalArgumentException
46      {
47          try
48          {
49              setValue( InetAddress.getByName( text ) );
50          }
51          catch( UnknownHostException uhe )
52          {
53              throw new IllegalArgumentException( uhe );
54          }
55      }
56  }
