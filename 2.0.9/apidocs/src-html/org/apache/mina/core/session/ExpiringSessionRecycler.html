020package org.apache.mina.core.session;
021
022import java.net.SocketAddress;
023
024import org.apache.mina.util.ExpirationListener;
025import org.apache.mina.util.ExpiringMap;
026
027/**
028 * An {@link IoSessionRecycler} with sessions that time out on inactivity.
029 *
030 * TODO Document me.
031 *
032 * @author <a href="http://mina.apache.org">Apache MINA Project</a>
033 * @org.apache.xbean.XBean
034 */
035public class ExpiringSessionRecycler implements IoSessionRecycler {
036    private ExpiringMap<SocketAddress, IoSession> sessionMap;
037
038    private ExpiringMap<SocketAddress, IoSession>.Expirer mapExpirer;
039
040    public ExpiringSessionRecycler() {
041        this(ExpiringMap.DEFAULT_TIME_TO_LIVE);
042    }
043
044    public ExpiringSessionRecycler(int timeToLive) {
045        this(timeToLive, ExpiringMap.DEFAULT_EXPIRATION_INTERVAL);
046    }
047
048    public ExpiringSessionRecycler(int timeToLive, int expirationInterval) {
049        sessionMap = new ExpiringMap<SocketAddress, IoSession>(timeToLive, expirationInterval);
050        mapExpirer = sessionMap.getExpirer();
051        sessionMap.addExpirationListener(new DefaultExpirationListener());
052    }
053
054    public void put(IoSession session) {
055        mapExpirer.startExpiringIfNotStarted();
056
057        SocketAddress key = session.getRemoteAddress();
058
059        if (!sessionMap.containsKey(key)) {
060            sessionMap.put(key, session);
061        }
062    }
063
064    public IoSession recycle(SocketAddress remoteAddress) {
065        return sessionMap.get(remoteAddress);
066    }
067
068    public void remove(IoSession session) {
069        sessionMap.remove(session.getRemoteAddress());
070    }
071
072    public void stopExpiring() {
073        mapExpirer.stopExpiring();
074    }
075
076    public int getExpirationInterval() {
077        return sessionMap.getExpirationInterval();
078    }
079
080    public int getTimeToLive() {
081        return sessionMap.getTimeToLive();
082    }
083
084    public void setExpirationInterval(int expirationInterval) {
085        sessionMap.setExpirationInterval(expirationInterval);
086    }
087
088    public void setTimeToLive(int timeToLive) {
089        sessionMap.setTimeToLive(timeToLive);
090    }
091
092    private class DefaultExpirationListener implements ExpirationListener<IoSession> {
093        public void expired(IoSession expiredSession) {
094            expiredSession.close(true);
095        }
096    }
097}
