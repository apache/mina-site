001package org.apache.mina.http;
002
003import java.nio.ByteBuffer;
004import java.nio.charset.Charset;
005import java.nio.charset.CharsetEncoder;
006import java.util.Map;
007
008import org.apache.mina.core.buffer.IoBuffer;
009import org.apache.mina.core.session.IoSession;
010import org.apache.mina.filter.codec.ProtocolEncoder;
011import org.apache.mina.filter.codec.ProtocolEncoderOutput;
012import org.apache.mina.http.api.HttpEndOfContent;
013import org.apache.mina.http.api.HttpRequest;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017public class HttpClientEncoder implements ProtocolEncoder {
018    private static final Logger LOG = LoggerFactory.getLogger(HttpClientCodec.class);
019    private static final CharsetEncoder ENCODER = Charset.forName("UTF-8").newEncoder();
020
021        public void encode(IoSession session, Object message, ProtocolEncoderOutput out)
022                        throws Exception {
023                LOG.debug("encode {}", message.getClass().getCanonicalName());
024                if (message instanceof HttpRequest) {
025                        LOG.debug("HttpRequest");
026                        HttpRequest msg = (HttpRequest)message;
027                        StringBuilder sb = new StringBuilder(msg.getMethod().toString());
028                        sb.append(" ");
029                        sb.append(msg.getRequestPath());
030                        if (!"".equals(msg.getQueryString())) {
031                                sb.append("?");
032                                sb.append(msg.getQueryString());
033                        }
034                        sb.append(" ");
035                        sb.append(msg.getProtocolVersion());
036                        sb.append("\r\n");
037
038                        for (Map.Entry<String, String> header : msg.getHeaders().entrySet()) {
039                sb.append(header.getKey());
040                sb.append(": ");
041                sb.append(header.getValue());
042                sb.append("\r\n");
043            }
044            sb.append("\r\n");
045            // Java 6 >> byte[] bytes = sb.toString().getBytes(Charset.forName("UTF-8"));
046            // byte[] bytes = sb.toString().getBytes();
047            // out.write(ByteBuffer.wrap(bytes));
048            IoBuffer buf = IoBuffer.allocate(sb.length()).setAutoExpand(true);
049            buf.putString(sb.toString(), ENCODER);
050            buf.flip();
051            out.write(buf);
052        } else if (message instanceof ByteBuffer) {
053                LOG.debug("Body");
054                out.write(message);
055        } else if (message instanceof HttpEndOfContent) {
056                LOG.debug("End of Content");
057            // end of HTTP content
058            // keep alive ?
059                }
060
061        }
062
063        public void dispose(IoSession arg0) throws Exception {
064                // TODO Auto-generated method stub
065
066        }
067
068}
