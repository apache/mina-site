020package org.apache.mina.proxy.handlers.socks;
021
022import java.util.Arrays;
023
024import org.apache.mina.core.buffer.IoBuffer;
025import org.apache.mina.core.filterchain.IoFilter.NextFilter;
026import org.apache.mina.proxy.session.ProxyIoSession;
027import org.apache.mina.proxy.utils.ByteUtilities;
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030
031/**
032 * Socks4LogicHandler.java - SOCKS4/SOCKS4a authentication mechanisms logic handler.
033 * 
034 * @author <a href="http://mina.apache.org">Apache MINA Project</a>
035 * @since MINA 2.0.0-M3
036 */
037public class Socks4LogicHandler extends AbstractSocksLogicHandler {
038
039    private final static Logger logger = LoggerFactory.getLogger(Socks4LogicHandler.class);
040
041    /**
042     * @see AbstractSocksLogicHandler#AbstractSocksLogicHandler(ProxyIoSession)
043     */
044    public Socks4LogicHandler(final ProxyIoSession proxyIoSession) {
045        super(proxyIoSession);
046    }
047
048    /**
049     * Perform the handshake.
050     * 
051     * @param nextFilter the next filter
052     */
053    public void doHandshake(final NextFilter nextFilter) {
054        logger.debug(" doHandshake()");
055
056        // Send request
057        writeRequest(nextFilter, request);
058    }
059
060    /**
061     * Encode a SOCKS4/SOCKS4a request and writes it to the next filter
062     * so it can be sent to the proxy server.
063     * 
064     * @param nextFilter the next filter
065     * @param request the request to send.
066     */
067    protected void writeRequest(final NextFilter nextFilter, final SocksProxyRequest request) {
068        try {
069            boolean isV4ARequest = Arrays.equals(request.getIpAddress(), SocksProxyConstants.FAKE_IP);
070            byte[] userID = request.getUserName().getBytes("ASCII");
071            byte[] host = isV4ARequest ? request.getHost().getBytes("ASCII") : null;
072
073            int len = 9 + userID.length;
074
075            if (isV4ARequest) {
076                len += host.length + 1;
077            }
078
079            IoBuffer buf = IoBuffer.allocate(len);
080
081            buf.put(request.getProtocolVersion());
082            buf.put(request.getCommandCode());
083            buf.put(request.getPort());
084            buf.put(request.getIpAddress());
085            buf.put(userID);
086            buf.put(SocksProxyConstants.TERMINATOR);
087
088            if (isV4ARequest) {
089                buf.put(host);
090                buf.put(SocksProxyConstants.TERMINATOR);
091            }
092
093            if (isV4ARequest) {
094                logger.debug("  sending SOCKS4a request");
095            } else {
096                logger.debug("  sending SOCKS4 request");
097            }
098
099            buf.flip();
100            writeData(nextFilter, buf);
101        } catch (Exception ex) {
102            closeSession("Unable to send Socks request: ", ex);
103        }
104    }
105
106    /**
107     * Handle incoming data during the handshake process. Should consume only the
108     * handshake data from the buffer, leaving any extra data in place.
109     * 
110     * @param nextFilter the next filter
111     * @param buf the server response data buffer
112     */
113    public void messageReceived(final NextFilter nextFilter, final IoBuffer buf) {
114        try {
115            if (buf.remaining() >= SocksProxyConstants.SOCKS_4_RESPONSE_SIZE) {
116                handleResponse(buf);
117            }
118        } catch (Exception ex) {
119            closeSession("Proxy handshake failed: ", ex);
120        }
121    }
122
123    /**
124     * Handle a SOCKS4/SOCKS4a response from the proxy server. Test
125     * the response buffer reply code and call {@link #setHandshakeComplete()}
126     * if access is granted.
127     * 
128     * @param buf the buffer holding the server response data.
129     * @throws exception if server response is malformed or if request is rejected
130     * by the proxy server.
131     */
132    protected void handleResponse(final IoBuffer buf) throws Exception {
133        byte first = buf.get(0);
134
135        if (first != 0) {
136            throw new Exception("Socks response seems to be malformed");
137        }
138
139        byte status = buf.get(1);
140
141        // Consumes all the response data from the buffer
142        buf.position(buf.position() + SocksProxyConstants.SOCKS_4_RESPONSE_SIZE);
143
144        if (status == SocksProxyConstants.V4_REPLY_REQUEST_GRANTED) {
145            setHandshakeComplete();
146        } else {
147            throw new Exception("Proxy handshake failed - Code: 0x" + ByteUtilities.asHex(new byte[] { status }) + " ("
148                    + SocksProxyConstants.getReplyCodeAsString(status) + ")");
149        }
150    }
151}
